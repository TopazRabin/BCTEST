    PROCEDURE TestAccountNo@1(AccountNo@1000000000 : Code[20]);
    VAR
      i@1240030001 : Integer;
      FoundNonZero@1240030002 : Boolean;
    BEGIN
      IF AccountNo = '' THEN
        EXIT;

      REPEAT
        i := i + 1;
        IF STRPOS('AEFKMNRTVWXY0123456789',COPYSTR(AccountNo,i,1)) = 0 THEN
          ERROR(Text005);
        IF COPYSTR(AccountNo,i,1) <> '0' THEN
          FoundNonZero := TRUE;
      UNTIL i = STRLEN(AccountNo);

      IF NOT FoundNonZero THEN
        ERROR(Text006);
    END;

    BEGIN
    END.
