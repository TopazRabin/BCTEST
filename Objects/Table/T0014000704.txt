    PROCEDURE PackageLevel@3() : Integer;
    VAR
      MinimumLevel@1000000000 : Integer;
      TempLevel@1000000001 : Integer;
    BEGIN
      IF "Original Package No." <> '' THEN BEGIN
        IF NOT PostedPackage.GET("Original Package No.") THEN
          EXIT(0);
      END ELSE
        PostedPackage := Rec;

      PostedPackageLine.RESET;
      PostedPackageLine.SETRANGE("Package No.",PostedPackage."No.");
      PostedPackageLine.SETRANGE(
        Type,PostedPackageLine.Type::"Account (G/L)",PostedPackageLine.Type::Resource);
      PostedPackageLine.SETFILTER("No.",'<>%1','');
      IF PostedPackageLine.FIND('-') THEN BEGIN
        EXIT(1);
      END ELSE BEGIN
        PostedPackageLine.SETRANGE(Type,PostedPackageLine.Type::Package);
        IF PostedPackageLine.FIND('-') THEN BEGIN
          REPEAT
            IF NOT PostedPackage.GET(PostedPackageLine."No.") THEN
              EXIT(0);

            TempLevel := PostedPackage.PackageLevel;
            IF MinimumLevel = 0 THEN
              MinimumLevel := TempLevel
            ELSE
              IF TempLevel < MinimumLevel THEN
                MinimumLevel := TempLevel;

            IF MinimumLevel = 1 THEN
              EXIT(2);
          UNTIL PostedPackageLine.NEXT = 0;
            EXIT(MinimumLevel + 1);
        END ELSE
          EXIT(1);
      END;
    END;
