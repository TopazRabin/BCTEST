    LOCAL PROCEDURE getCustomerOptions@1000000000();
    VAR
      lrecLPHeader@1000000000 : Record 23044505;
      lcuPackMgmt@1000000001 : Codeunit 23044601;
      loptDocType@1000000002 : Option;
    BEGIN
      //<FUNC>
      //  Description: Apply the customer options template (if any)
      //  Called From: This table (T23044608)
      //  Side Effect: None
      //</FUNC>

      lrecLPHeader.SETAUTOCALCFIELDS("Source Document", "Source No.");
      IF ( (NOT lrecLPHeader.GET(Rec."License Plate No.")) OR
           (lrecLPHeader."Source Document" = lrecLPHeader."Source Document"::" ") )
      THEN
        EXIT;

      lcuPackMgmt.getDShipDocTypeFromLPSrcType(lrecLPHeader."Source Document", loptDocType);
      bFromInsert := TRUE;
      updateFromCustomerTemplate(loptDocType, lrecLPHeader."Source No.", TRUE);
      //lcuPackMgmt.updatePackOptionsFromCustomerTemplate(Rec, loptDocType, lrecLPHeader."Source No.", TRUE);
    END;

    PROCEDURE updateFromCustomerTemplate@1000000001(poptDocType@1000000000 : Option;pcodDocNo@1000000001 : Code[20];pbForce@1000000007 : Boolean);
    VAR
      lrecSalesHeader@1000000006 : Record 36;
      lrecCustomerOptions@1000000005 : Record 23044614;
      lrecPackageOptionsTemplate@1000000004 : Record 23044608;
      lcuPackMgmt@1000000003 : Codeunit 23044601;
      lbTemplateFound@1000000002 : Boolean;
    BEGIN
      //<FUNC>
      //  Description: Applies the customer options template
      //  Called From: This table (T23044608), CU23044601
      //  Side Effect: None
      //</FUNC>

      IF ( (Rec."Entry Type" = Rec."Entry Type"::fromtemplate) AND (NOT pbForce) ) THEN
        EXIT;

      IF ( NOT lcuPackMgmt.getSalesHeader(poptDocType, pcodDocNo, lrecSalesHeader) ) THEN
        EXIT;

      IF ( lcuPackMgmt.getCustomerOptions(lrecSalesHeader, lrecCustomerOptions) ) THEN BEGIN
        lrecPackageOptionsTemplate.SETRANGE("License Plate No.", lrecCustomerOptions."Package Options Template Code");
        lbTemplateFound := lrecPackageOptionsTemplate.FINDFIRST;
        IF ( lbTemplateFound ) THEN BEGIN
          IF ( (pbForce) OR (Rec."Entry Type" = Rec."Entry Type"::standard) ) THEN
            applyTemplate(lrecPackageOptionsTemplate);
        END;
      END;
    END;

    PROCEDURE applyTemplate@1000000003(precPackageOptionsTemplate@1000000000 : Record 23044608);
    VAR
      lcodLPNo@1000000001 : Code[20];
    BEGIN
      //<FUNC>
      //  Description: Applies the given template
      //  Called From: This table (T23044608)
      //  Side Effect: None
      //</FUNC>

      Rec.TRANSFERFIELDS(precPackageOptionsTemplate, FALSE);
      Rec."Entry Type" := Rec."Entry Type"::fromtemplate;
      IF ( NOT bFromInsert ) THEN
        Rec.MODIFY(TRUE);
    END;
