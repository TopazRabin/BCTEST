  PROPERTIES
  {
    OnInsert=BEGIN
               CASE Type OF
                 Type::Package:
                   BEGIN
                     Package.TESTFIELD("No.");

                     GetShippingSetup;
                     ExportDocSetup.GET;

                     DocumentOptionPageFound := FALSE;
                     IF Package."Source ID" <> '' THEN
                       IF UPSOptionPage.GET(
                            UPSOptionPage.Type::Document,Package."Source ID",
                            Package."Source Type",Package."Source Subtype")
                       THEN BEGIN
                         DocumentOptionPageFound := TRUE;

                         Rec := UPSOptionPage;
                       END;

                     "Shipping Agent Code" := Package."Shipping Agent Code";
                     "Shipping Agent Service" := Package."Shipping Agent Service";
                     "World Wide Service" := Package."World Wide Service";

                     IF NOT DocumentOptionPageFound THEN BEGIN
                       UPSOptionPage.RESET;
                       UPSOptionPage.SETCURRENTKEY(
                         Type,"Shipping Agent Code","Shipping Agent Service","World Wide Service");
                       UPSOptionPage.SETRANGE(Type,UPSOptionPage.Type::Setup);
                       UPSOptionPage.SETRANGE("Shipping Agent Code",Package."Shipping Agent Code");
                       UPSOptionPage.SETRANGE("Shipping Agent Service",Package."Shipping Agent Service");
                       UPSOptionPage.SETRANGE("World Wide Service",Package."World Wide Service");
                       IF UPSOptionPage.FIND('-') THEN
                         Rec := UPSOptionPage;

                       IF Package."Source Type" <> 0 THEN BEGIN
                         PackingControl.TransferFromSource(
                           Package."Source Type",Package."Source Subtype",Package."Source ID");

                         TransferFromPackingControl(PackingControl);
                       END;
                     END;

                     Type := Type::Package;
                     "Source ID" := Package."No.";
                     "Source Type" := 0;
                     "Source Subtype" := 0;
                     "Second Source ID" := '';

                     IF (Package."Ship-to No." <> '') AND
                        ((Package."Source ID" = '') OR NOT DocumentOptionPageFound)
                     THEN BEGIN
                       CLEAR(UPSOptionPage);
                       CASE TRUE OF
                         Package."Ship-to Type" = Package."Ship-to Type"::Customer:
                           IF UPSOptionPage.GET(
                                UPSOptionPage.Type::"Master Data",Package."Ship-to No.",DATABASE::Customer,
                                0,Package."Ship-to Code")
                           THEN
                             ;
                         Package."Ship-to Type" = Package."Ship-to Type"::Vendor:
                           IF UPSOptionPage.GET(
                                UPSOptionPage.Type::"Master Data",Package."Ship-to No.",DATABASE::Vendor,
                                0,Package."Ship-to Code")
                           THEN
                             ;
                         Package."Ship-to Type" = Package."Ship-to Type"::Location:
                           IF UPSOptionPage.GET(
                                UPSOptionPage.Type::"Master Data",Package."Ship-to No.",DATABASE::Location,
                                0,'')
                           THEN
                             ;
                       END;

                       IF UPSOptionPage."Source ID" <> '' THEN
                         MergeOptionPage(UPSOptionPage);
                     END;

                     IF GetShippingAccount(
                          Package."Ship-to Type",Package."Ship-to No.",Package."Ship-to Code",
                          Package."Shipping Agent Code",Package."Third Party Ship. Account No.")
                     THEN
                       TransferFromShippingAccount(ShippingAccount);

                     IF Package."Ship-to Country Code" <> '' THEN BEGIN
                       Country.GET(Package."Ship-to Country Code");
                       IF EShipValidation."Country/Region-UPSWorldWideService"(Country) OR
                          EShipValidation."Country/Region-UPSCanadianService"(Country) OR
                          EShipValidation."Country/Region-UPSPuertoRicoService"(Country)
                       THEN BEGIN
                         Country.TESTFIELD("ISO 3 digit Country Code");

                         "Include Commodity Information" := TRUE;
                         IF ExportDocSetup."Print NAFTA Cert. of Origin" AND
                            ((Country."ISO 3 digit Country Code" = '124') OR
                             (Country."ISO 3 digit Country Code" = '484'))
                         THEN
                           "Certificate of Origin" := "Certificate of Origin"::"NAFTA Included";

                         IF ExportDocSetup."Print U.S. Cert. of Origin" AND
                            (Country."ISO 3 digit Country Code" <> '124') AND
                            (Country."ISO 3 digit Country Code" <> '484')
                         THEN
                           "Certificate of Origin" := "Certificate of Origin"::"Non-NAFTA Included";
                       END;
                     END;
                   END;
                 Type::Document:
                   BEGIN
                     TESTFIELD("Source ID");
                     TESTFIELD("Source Type");

                     PackingControl.TransferFromSource("Source Type","Source Subtype","Source ID");
                     TransferFromPackingControl(PackingControl);

                     IF GetShippingAccount(
                          PackingControl."Ship-to Type",PackingControl."Ship-to No.",PackingControl."Ship-to Code",
                          PackingControl."Shipping Agent Code",PackingControl."Third Party Ship. Account No.")
                     THEN
                       TransferFromShippingAccount(ShippingAccount);
                   END;
                 Type::"Bill of Lading":
                   BEGIN
                     TESTFIELD("Source ID");

                     BillOfLading.GET("Source ID");
                     BillOfLading.TESTFIELD(Released,FALSE);
                     TransferFromBillOfLading(BillOfLading);

                     IF GetShippingAccount(
                          BillOfLading."Ship-to Type",BillOfLading."Ship-to No.",BillOfLading."Ship-to Code",
                          BillOfLading."Shipping Agent Code",BillOfLading."Third Party Ship. Account No.")
                     THEN
                       TransferFromShippingAccount(ShippingAccount);
                   END;
                 Type::"Master Data":
                    BEGIN
                     TESTFIELD("Source ID");
                     TESTFIELD("Source Type");
                    END;
                 ELSE
                   TESTFIELD("Source ID");
               END;
             END;

    OnModify=BEGIN
               CASE Type OF
                 Type::Package:
                   BEGIN
                     IF Package.GET("Source ID") THEN ;
                     Package.TESTFIELD(Closed,FALSE);
                     IF "Additional Handling Required" OR
                        "Hold At Location" OR
                        "Dangerous Goods" OR
                        "Dry Ice" OR
                        "Extended Area Surcharge" OR
                        "Verbal Confirm. of Delivery"
                     THEN
                       "Accessorial Record Required" := TRUE
                     ELSE
                       "Accessorial Record Required" := FALSE;
                     IF "Ship Notification Type" <> "Ship Notification Type"::None THEN BEGIN
                       "Ship Notification" := TRUE;
                       IF "UPS Advisory Info. Level" = "UPS Advisory Info. Level"::" " THEN
                         "UPS Advisory Info. Level" := "UPS Advisory Info. Level"::Shipment;
                     END;
                   END;
                 Type::"Bill of Lading":
                   BEGIN
                     BillOfLading.GET("Source ID");
                     BillOfLading.TESTFIELD(Released,FALSE);
                   END;
               END;
             END;

    OnDelete=VAR
               Shipping@1240020000 : Codeunit 14000701;
               PackageReturn@1240020001 : Record 14000701;
               UPSOptionPage2@1240020002 : Record 14000761;
             BEGIN
               CASE Type OF
                 Type::Package:
                   BEGIN
                     IF Package.GET("Source ID") THEN ;
                     Package.TESTFIELD(Closed,FALSE);

                     IF "Ship and Return Label Pkg No." <> '' THEN BEGIN
                       PackageReturn.RESET;
                       PackageReturn.SETRANGE("No.","Ship and Return Label Pkg No.");
                       IF PackageReturn.FIND('-') THEN BEGIN
                         IF PackageReturn.Closed THEN
                           Shipping.OpenPackage(PackageReturn);
                         PackageReturn.DELETE(TRUE);
                       END;
                     END;
                   END;
                 Type::"Bill of Lading":
                   BEGIN
                     BillOfLading.GET("Source ID");
                     BillOfLading.TESTFIELD(Released,FALSE);
                   END;
               END;
             END;

    LOCAL PROCEDURE GetShippingAgentAccount@1007();
    BEGIN
      IF Type = Type::Package THEN BEGIN
        Package.GET("Source ID");

        PackingStation.GetPackingStation;

      END;
    END;

    PROCEDURE TestHMU@1240030002(HMURequired@1240030000 : Boolean;HMUVersion@1240030002 : ' ,2.00,5.05,7.00,SPF 0200';FieldCaption@1240030001 : Text[250]);
    BEGIN
      GetShippingSetup;

      IF HMURequired THEN
        ShippingSetup.TESTFIELD("UPS Version",ShippingSetup."UPS Version"::HMU);

      IF ShippingSetup."UPS Version" = ShippingSetup."UPS Version"::HMU THEN BEGIN
        GetShippingAgentAccount;
        IF ShippingAgentAccount."Manifest Version" < HMUVersion THEN
          ERROR(
            Text002,FieldCaption,ShippingAgentAccount.FIELDCAPTION("Manifest Version"),
            FORMAT(HMUVersion));
      END;
    END;

    PROCEDURE TestUPSlink@1240030003();
