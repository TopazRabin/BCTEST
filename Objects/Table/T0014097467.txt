    PROCEDURE GetCaption@1240020002(CurrPeriod@1240020000 : Integer) : Text[80];
    VAR
      ForecastInventoryPeriod@1240020003 : Record 14000552;
      ItemUsageReview@1240020001 : Record 14097467;
      CapPeriod@1240020002 : Integer;
      OnPage@1240020004 : Boolean;
    BEGIN
      IF CurrPeriod = 1 THEN BEGIN
        CapPeriod := 0;
        ItemUsageReview.RESET;
        FILTERGROUP := 2;  // page group
        ItemUsageReview.SETFILTER("Batch Type", GETFILTER("Batch Type"));
        ItemUsageReview.SETFILTER("Batch Name", GETFILTER("Batch Name"));
        OnPage := TRUE;
        IF (ItemUsageReview.GETFILTER("Batch Name") = '') THEN BEGIN
          OnPage := FALSE;
          FILTERGROUP := 0;  // other source
          ItemUsageReview.SETFILTER("Batch Type", GETFILTER("Batch Type"));
          ItemUsageReview.SETFILTER("Batch Name", GETFILTER("Batch Name"));
        END ELSE
        FILTERGROUP := 0;
        IF ItemUsageReview.FIND('-') THEN BEGIN
          IF ItemUsageReview."Starting Date" <> 0D THEN BEGIN
            ForecastInventoryPeriod.RESET;
            ForecastInventoryPeriod.SETRANGE(Type,ForecastInventoryPeriod.Type::Calculate);
            ForecastInventoryPeriod.SETFILTER("Starting Date",'%1..',ItemUsageReview."Starting Date");
            IF ForecastInventoryPeriod.FIND('-') THEN
              REPEAT
                CapPeriod := CapPeriod + 1;
                StartingDateArray[CapPeriod] := ForecastInventoryPeriod."Starting Date";
              UNTIL (ForecastInventoryPeriod.NEXT = 0) OR
                    (CapPeriod = 52);
          END;
        END;
      END;

      IF LocInvPer."Starting Date" = 0D THEN BEGIN
        Location.SETRANGE("Enable Adv. Forecasting",TRUE);
        Location.FINDFIRST;
        LocInvPer.SETRANGE("Location Code",Location.Code);
        LocInvPer.SETRANGE("Period End Closed",TRUE);
        LocInvPer.FINDLAST;
      END;

      IF StartingDateArray[CurrPeriod] > LocInvPer."Starting Date" THEN BEGIN
        IF OnPage THEN
          EXIT('3,' + 'Statistical: ' + FORMAT(StartingDateArray[CurrPeriod]))
      ELSE
          EXIT('Statistical: ' + FORMAT(StartingDateArray[CurrPeriod]))
      END ELSE
        EXIT('3,' + 'Hist: ' + FORMAT(StartingDateArray[CurrPeriod]));
    END;
