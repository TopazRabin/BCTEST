    PROCEDURE AddToObjectProperty@1240520009(VAR ESObjectInfo@1240520008 : Record 14123509;AddAsLimited@1240520003 : Boolean;AskForQuickSecurityType@1240520001 : Boolean;ShowStatus@1240520013 : Boolean);
    VAR
      ESObjectProperty@1240520009 : Record 14123522;
      ESSecuritySetup@1240520005 : Record 14123521;
      TotalRecords@1240520011 : Integer;
      TotalAdded@1240520010 : Integer;
      QuickSecurityType@1240520002 : Integer;
      Window@1240520012 : Dialog;
    BEGIN
      IF AskForQuickSecurityType THEN BEGIN
        ESObjectInfo.FIND('-');
        IF ESObjectInfo."Object Type" = ESObjectInfo."Object Type"::TableData THEN BEGIN
          ESObjectInfo.FIND('+');
          IF ESObjectInfo."Object Type" > ESObjectInfo."Object Type"::TableData THEN
            ERROR(Text006);
        END;

        IF ESObjectInfo."Object Type" = ESObjectInfo."Object Type"::TableData THEN
          QuickSecurityType :=
            STRMENU(
              Text004,2,
              STRSUBSTNO(Text005,FIELDCAPTION("Quick Security Type")))
        ELSE
          QuickSecurityType :=
            STRMENU(
              Text007,3,
              STRSUBSTNO(Text005,FIELDCAPTION("Quick Security Type")));

        IF QuickSecurityType = 0 THEN
          EXIT;
        QuickSecurityType := QuickSecurityType - 1;
      END;

      IF GUIALLOWED THEN
        Window.OPEN(ESSecuritySetup.ExtendProgressText(Text001) + Text002);

      IF ESObjectInfo.FIND('-') THEN
        REPEAT
          IF GUIALLOWED THEN
            Window.UPDATE(1,STRSUBSTNO('%1 %2',ESObjectInfo."Object Type",ESObjectInfo."Object ID"));

          TotalRecords := TotalRecords + 1;

          IF AskForQuickSecurityType THEN
            IF NOT (QuickSecurityType IN
                    [ESObjectProperty."Quick Security Type"::"Full Access",
                     ESObjectProperty."Quick Security Type"::"No Access"])
            THEN
              ESObjectInfo.TESTFIELD("Object Type",ESObjectInfo."Object Type"::TableData);

          IF NOT ESObjectProperty.GET(ESObjectInfo."Object Type",ESObjectInfo."Object ID") THEN BEGIN
            ESObjectProperty.INIT;
            ESObjectProperty."Object Type" := ESObjectInfo."Object Type";
            ESObjectProperty."Object ID" := ESObjectInfo."Object ID";
            ESObjectProperty.InitObjectLevelPermissions;
            IF AskForQuickSecurityType THEN
              ESObjectProperty."Quick Security Type" := QuickSecurityType
            ELSE
              ESObjectProperty."Object Level Permissions" := AddAsLimited;
            ESObjectProperty.SetNoInitObjectLevel(TRUE);
            ESObjectProperty.INSERT(TRUE);

            TotalAdded := TotalAdded + 1;
          END ELSE
            IF AskForQuickSecurityType THEN BEGIN
              ESObjectProperty."Quick Security Type" := QuickSecurityType;
              ESObjectProperty.MODIFY(TRUE);
            END ELSE BEGIN
              IF ESObjectProperty."Object Level Permissions" <> AddAsLimited THEN BEGIN
                ESObjectProperty."Object Level Permissions" := AddAsLimited;
                ESObjectProperty.MODIFY(TRUE);
              END;
            END;
        UNTIL ESObjectInfo.NEXT = 0;

      IF GUIALLOWED THEN
        Window.CLOSE;

      IF ShowStatus THEN
        MESSAGE(Text003,TotalAdded,TotalRecords - TotalAdded);
    END;
