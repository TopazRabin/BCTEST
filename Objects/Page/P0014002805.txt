    LOCAL PROCEDURE PrintItemLabelforInner@1000000001();
    VAR
      SalesLineLoc@1000000000 : Record 37;
      ItemLoc@1000000001 : Record 27;
      ItemUnitofMeasure@1000000002 : Record 5404;
      ItemMasterInnerLabel@1000000003 : Report 51476;
      LabelPrintedLoc@1000000004 : Boolean;
      PackingStation@1000000006 : Record 14000709;
      UserSetup@1000000005 : Record 91;
      WarehouseRequest@1000000007 : Record 5765;
      WarehouseActivityLine@1000000008 : Record 5767;
    BEGIN
      //TPZ3264
      LabelPrintedLoc := FALSE;
      IF UserSetup.GET(USERID) THEN
        IF PackingStation.GET(UserSetup."Packing Station") AND PackingStation."Print Inner Label" THEN
          WarehouseRequest.RESET;
          CurrPage.SETSELECTIONFILTER(WarehouseRequest);
          IF WarehouseRequest.FINDFIRST THEN
            REPEAT
              IF WarehouseRequest."Source Document" = WarehouseRequest."Source Document"::"Sales Order" THEN BEGIN
                WarehouseRequest.CALCFIELDS("Put-away / Pick No.");
                IF WarehouseRequest."Put-away / Pick No." <> '' THEN BEGIN
                  WarehouseActivityLine.RESET;
                  WarehouseActivityLine.SETRANGE("No.",WarehouseRequest."Put-away / Pick No.");
                  WarehouseActivityLine.SETRANGE("Action Type",WarehouseActivityLine."Action Type"::Place);
                  IF WarehouseActivityLine.FINDFIRST THEN
                    REPEAT
            SalesLineLoc.RESET;
            SalesLineLoc.SETRANGE("Document Type",SalesLineLoc."Document Type"::Order);
                      SalesLineLoc.SETRANGE("Document No.",WarehouseActivityLine."Source No.");
                      SalesLineLoc.SETRANGE("Line No.",WarehouseActivityLine."Source Line No.");
            SalesLineLoc.SETRANGE(Type,SalesLineLoc.Type::Item);
                      SalesLineLoc.SETRANGE("No.",WarehouseActivityLine."Item No.");
            IF SalesLineLoc.FINDFIRST THEN
                IF ItemUnitofMeasure.GET(SalesLineLoc."No.",'I') AND
                  (SalesLineLoc.Quantity < ItemUnitofMeasure."Qty. per Unit of Measure") THEN BEGIN
                  LabelPrintedLoc := TRUE;
                  CreatePrinterSelection;
                  CLEAR(ItemMasterInnerLabel);
                          ItemMasterInnerLabel.Setfilter(SalesLineLoc."Location Code",1,SalesLineLoc.Quantity,SalesLineLoc."No.",SalesLineLoc."Document No.");
                  ItemMasterInnerLabel.USEREQUESTPAGE(FALSE);
                  ItemMasterInnerLabel.RUN;
                  DeletePrinterSelection;
                END;
                    UNTIL WarehouseActivityLine.NEXT = 0;
                END;
          END;
            UNTIL WarehouseRequest.NEXT = 0;
      IF (fromprintaction AND  NOT LabelPrintedLoc ) THEN MESSAGE('Nothing to print');
      fromprintaction := FALSE;
    END;
