    PROCEDURE SetFields@1();
    BEGIN
      OrderQtyMethodEditable := FALSE;
      MinMaxShelfEditable := TRUE;
      MinMaxFieldsEditable := FALSE;
      LeadTimeFieldsEditable := FALSE;
      SafetyStockDaysFieldsEditable := TRUE;
      SafetyStockQtyFieldsEditable := TRUE;
      SafetyStockFieldsEditable := TRUE;
      CASE "Replenishment Source Type" OF
          "Replenishment Source Type"::Location:
              BEGIN
                  MinMaxShelfEditable := TRUE;
                  CASE "Replenishment Model" OF
                      "Replenishment Model"::"Min/Max":
                          BEGIN
                              MinMaxFieldsEditable := TRUE;
                              SafetyStockFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := FALSE;
                              SafetyStockQtyFieldsEditable := FALSE;
                              IF "Minimum Inventory" > 0 THEN
                                MinMaxShelfEditable := FALSE;
                          END;
                      ELSE BEGIN
                              MinMaxFieldsEditable := FALSE;
                          END;
                  END;
              END;
          "Replenishment Source Type"::Vendor:
              BEGIN
                  MinMaxShelfEditable := TRUE;
                  OrderQtyMethodEditable := TRUE;
                  LeadTimeFieldsEditable := TRUE;
                  CASE "Replenishment Model" OF
                      "Replenishment Model"::"Min/Max":
                          BEGIN
                              MinMaxFieldsEditable := TRUE;
                              SafetyStockFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := FALSE;
                              SafetyStockQtyFieldsEditable := FALSE;
                              IF "Minimum Inventory" > 0 THEN
                                MinMaxShelfEditable := FALSE;
                          END;
                      ELSE BEGIN
                              MinMaxFieldsEditable := FALSE;
                          END;
                  END;
              END;
          "Replenishment Source Type"::Production:
              BEGIN
                  CASE "Replenishment Model" OF
                      "Replenishment Model"::"Min/Max":
                          BEGIN
                              MinMaxFieldsEditable := TRUE;
                              SafetyStockFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := FALSE;
                              SafetyStockQtyFieldsEditable := FALSE;
                              IF "Minimum Inventory" > 0 THEN
                                MinMaxShelfEditable := FALSE;
                          END;
                      ELSE BEGIN
                              MinMaxFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := TRUE;
                          END;
                  END;
              END;
          "Replenishment Source Type"::Assembly:
              BEGIN
                  CASE "Replenishment Model" OF
                      "Replenishment Model"::"Min/Max":
                          BEGIN
                              MinMaxFieldsEditable := TRUE;
                              SafetyStockFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := FALSE;
                              SafetyStockQtyFieldsEditable := FALSE;
                              IF "Minimum Inventory" > 0 THEN
                                MinMaxShelfEditable := FALSE;
                          END;
                      ELSE BEGIN
                              MinMaxFieldsEditable := FALSE;
                              SafetyStockDaysFieldsEditable := TRUE;
                          END;
                  END;
              END;
      END;
      SummaryEditable := FALSE;
      IF ("Variant Code" = '') AND SummaryVisible THEN BEGIN
        SummaryEditable := TRUE;
      END;
    END;
