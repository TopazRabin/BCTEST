    [External]
    PROCEDURE SetTableViewFilters@1240020001();
    VAR
      RFEmployeeSetup@1240020000 : Record 14002802;
    BEGIN
      IF OutboundWhseRequestFilter."Location Code" = '' THEN
        SETRANGE("Location Code")
      ELSE
        SETFILTER("Location Code",OutboundWhseRequestFilter."Location Code");

      RFEmployeeSetup.GetRFEmployeeSetup(USERID);
      IF RFEmployeeSetup."Disable OWR Code Table Filters" THEN
        EXIT;

      IF OutboundWhseRequestFilter."Source Type Filter" = 0 THEN
        SETRANGE("Source Type")
      ELSE
        SETFILTER("Source Type",'%1',OutboundWhseRequestFilter."Source Type Filter");

      IF OutboundWhseRequestFilter."Document Status Filter" =
         OutboundWhseRequestFilter."Document Status Filter"::All
      THEN
        SETRANGE("Document Status")
      ELSE BEGIN
        CASE OutboundWhseRequestFilter."Document Status Filter" OF
          OutboundWhseRequestFilter."Document Status Filter"::Open:
            SETRANGE("Document Status","Document Status"::Open);
          OutboundWhseRequestFilter."Document Status Filter"::Released:
            SETRANGE("Document Status","Document Status"::Released);
        END;
      END;

      IF OutboundWhseRequestFilter."Shipment Method Code Filter" = '' THEN
        SETRANGE("Shipment Method Code")
      ELSE
        SETFILTER("Shipment Method Code",OutboundWhseRequestFilter."Shipment Method Code Filter");

      IF OutboundWhseRequestFilter."Shipping Agent Code Filter" = '' THEN
        SETRANGE("Shipping Agent Code")
      ELSE
        SETFILTER("Shipping Agent Code",OutboundWhseRequestFilter."Shipping Agent Code Filter");

      IF OutboundWhseRequestFilter."Shipping Advice Filter" =
         OutboundWhseRequestFilter."Shipping Advice Filter"::All
      THEN
        SETRANGE("Shipping Advice")
      ELSE BEGIN
        CASE OutboundWhseRequestFilter."Shipping Advice Filter" OF
          OutboundWhseRequestFilter."Shipping Advice Filter"::Partial:
            SETRANGE("Shipping Advice","Shipping Advice"::Partial);
          OutboundWhseRequestFilter."Shipping Advice Filter"::Complete:
            SETRANGE("Shipping Advice","Shipping Advice"::Complete);
        END;
      END;

      IF OutboundWhseRequestFilter."Destination No. Filter" = '' THEN
        SETRANGE("Destination No.")
      ELSE
        SETFILTER("Destination No.",OutboundWhseRequestFilter."Destination No. Filter");

      IF OutboundWhseRequestFilter."Action Code Filter" = '' THEN
        SETRANGE("Action Code")
      ELSE
        SETFILTER("Action Code",OutboundWhseRequestFilter."Action Code Filter");

      IF OutboundWhseRequestFilter."Action Responsible Filter" = '' THEN
        SETRANGE("Action Responsible")
      ELSE
        SETFILTER("Action Responsible",OutboundWhseRequestFilter."Action Responsible Filter");

      IF OutboundWhseRequestFilter."E-Ship Agent Service Filter" = '' THEN
        SETRANGE("E-Ship Agent Service")
      ELSE
        SETFILTER("E-Ship Agent Service",OutboundWhseRequestFilter."E-Ship Agent Service Filter");

      IF OutboundWhseRequestFilter."Shipment Date Filter" = '' THEN
        SETRANGE("Shipment Date")
      ELSE
        SETFILTER("Shipment Date",OutboundWhseRequestFilter.GetShipmentDateFilter(WORKDATE));

      IF OutboundWhseRequestFilter."EDI Order Filter" =
         OutboundWhseRequestFilter."EDI Order Filter"::All
      THEN
        SETRANGE("EDI Order")
      ELSE BEGIN
        CASE OutboundWhseRequestFilter."EDI Order Filter" OF
          OutboundWhseRequestFilter."EDI Order Filter"::"EDI Only":
            SETRANGE("EDI Order",TRUE);
          OutboundWhseRequestFilter."EDI Order Filter"::"No EDI Orders":
            SETRANGE("EDI Order",FALSE);
        END;
      END;

      IF OutboundWhseRequestFilter."ATO Pick Required Filter" =
         OutboundWhseRequestFilter."ATO Pick Required Filter"::All
      THEN
        SETRANGE("ATO Pick Required")
      ELSE BEGIN
        CASE OutboundWhseRequestFilter."ATO Pick Required Filter" OF
          OutboundWhseRequestFilter."ATO Pick Required Filter"::"ATO Orders Only":
            SETRANGE("ATO Pick Required",TRUE);
          OutboundWhseRequestFilter."ATO Pick Required Filter"::"No ATO Orders":
            SETRANGE("ATO Pick Required",FALSE);
        END;
      END;
    END;

      RFSetup.GET;

      IF RFSetup."Enable OWR Replenishments" THEN

      IF RFSetup."Enable Combined Picks" THEN

      IF RFSetup."Enable Extra Picks" THEN
        ShowOWRExtraPickOptions := TRUE;

      IF RFSetup."Enable Priority Picks" THEN BEGIN
        ShowOWRPriorityPickOptions := TRUE;
        ShowOWRWhseActGroupsTab := TRUE;

      IF RFSetup."Enable Whse. Activity Groups" THEN BEGIN
        ShowOWRWhseActGroups := TRUE;
        ShowOWRWhseActGroupsTab := TRUE;
      END;

        ShowOWRPerZoneOptions := TRUE;

      IF RFSetup."Enable Pick-from Filters" THEN BEGIN
        ShowOWRPickFromFilters := TRUE;
        ShowOWRFiltersFastTab := TRUE;
      END;

      IF RFSetup."Enable Line Date Filters" THEN BEGIN
        ShowOWRLineDateFilters := TRUE;
        ShowOWRFiltersFastTab := TRUE;
      END;

      IF RFSetup."Enable Limits" THEN
        ShowOWRLimits := TRUE;

      OutboundWhseRequestMgt.EShipInstalledEvent(EShipInstalled);
      IF EShipInstalled THEN
        ShowEShipOptions := TRUE;
