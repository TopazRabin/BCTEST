    PROCEDURE getTranslatedCaption@1000000003() : Text[100];
    VAR
      lrecTranslationValue@1000000001 : Record 23044519;
      lcodRegion@1000000000 : Code[10];
    BEGIN
      //  Will set the caption to the translation value if possible
      //  Note: Caption is translated (again) on return to handheld
      lcodRegion := cuTranslationMgmt.getRegionCode();

        IF lrecTranslationValue.GET(COPYSTR(Caption,1,MAXSTRLEN(lrecTranslationValue.Code)),lcodRegion) THEN BEGIN
         EXIT(lrecTranslationValue.Translation);
        END;
      END;

      EXIT(Caption);
    END;
