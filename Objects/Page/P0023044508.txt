    BEGIN

    PROCEDURE SetRecFilters@1();
    BEGIN
      IF(optFormType = optFormType::Main) THEN
        trecFilterRec."Form ID" := trecFilterRec."Form ID"::Main
      ELSE
        trecFilterRec."Form ID" := trecFilterRec."Form ID"::"Custom Application";


      SETRANGE("Device Config ID", trecFilterRec."Config ID");
      SETRANGE("Form ID", trecFilterRec."Form ID");

      IF ( trecFilterRec."Form ID" = trecFilterRec."Form ID"::"Custom Application" ) THEN BEGIN
        SETRANGE("Application Screen", trecFilterRec."Application Screen");
        bShowWorkflowScreen := TRUE;
      END ELSE BEGIN
        SETRANGE("Application Screen");// clear any filter.
        bShowWorkflowScreen := FALSE;
      END;

      cuMenuMgmt.SortMenus( Rec);



      bReadyShowMenuConfig := ( trecFilterRec."Form ID" <> trecFilterRec."Form ID"::"Custom Application" )
                              OR ( ( trecFilterRec."Form ID" = trecFilterRec."Form ID"::"Custom Application" )
                                   AND
                                  ( trecFilterRec."Application Screen" <> ''  ) );
      SetSubPageFilters();

      CurrPage.UPDATE(FALSE);
    END;

    PROCEDURE SetSubPageFilters@1000000002();
    BEGIN
      // Introduced to simplify menu item configuration.
      bShowCustomText := ( Rec."Standard Action" = Rec."Standard Action"::"Custom Text" );
      bShowPromptBarcode := ( Rec."Standard Action" = Rec."Standard Action"::PromptToBarcode );
      bShowLookupBarcode := ( Rec."Standard Action" = Rec."Standard Action"::LookupToBarcode );
      bShowCustomQuantity := ( Rec."Standard Action" = Rec."Standard Action"::"Custom Qty" );
      bShowModule := ( Rec."Standard Action" = Rec."Standard Action"::"Start Application" );
    END;

    PROCEDURE getTranslatedCaption@1000000003() : Text[100];
    VAR
      lrecTranslationValue@1000000001 : Record 23044519;
      lcodRegion@1000000000 : Code[10];
    BEGIN
      //  Will set the caption to the translation value if possible
      //  Note: Caption is translated (again) on return to handheld
      lcodRegion := cuTranslationMgmt.getRegionCode();
      IF lcodRegion <> '' THEN BEGIN
        IF lrecTranslationValue.GET(COPYSTR(Caption,1,MAXSTRLEN(lrecTranslationValue.Code)),lcodRegion) THEN BEGIN
         EXIT(lrecTranslationValue.Translation);
        END;
      END;

      EXIT(Caption);
    END;
