    PROCEDURE setFormFilter@1000000000(pcodCountNo@1000000000 : Code[20];pcodSheetName@1000000001 : Code[20]);
    BEGIN
      codCurrentCountNo := pcodCountNo;
      codCurrentSheetName := pcodSheetName;

      applyFilter();
    END;

    LOCAL PROCEDURE applyFilter@1000000001();
    BEGIN
      FILTERGROUP(2);
      IF( codCurrentCountNo <> '' ) THEN BEGIN
        SETRANGE("Count No.",codCurrentCountNo);
      END
      ELSE BEGIN
        SETRANGE("Count No.");
      END;

      IF( (codCurrentSheetName <> '') AND (NOT bShowAllSheets) ) THEN BEGIN
        SETRANGE("Sheet Name",codCurrentSheetName);
      END
      ELSE BEGIN
        SETRANGE("Sheet Name");
      END;

      CurrPage.UPDATE(FALSE);

      FILTERGROUP(0);

      // set the label caption
      IF( bShowAllSheets ) THEN
        sCurrentSheet := tcAllSheets
      ELSE
        sCurrentSheet := STRSUBSTNO(tcCurrentSheet,codCurrentSheetName);
    END;
