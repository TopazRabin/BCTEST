    LOCAL PROCEDURE EnableControls@2();
    BEGIN
      ResetControls;

      CASE Step OF
        Step::Start:
          ShowStartStep;
        Step::Enable:
          ShowEnableStep;
        Step::Default:
          ShowDefaultSetupStep;
        Step::Settings:
          ShowSettingsStep;
        Step::Settings2:
          ShowSettingsStep2;
        Step::Finish:
          ShowFinishStep;
      END;
    END;

    LOCAL PROCEDURE FinishAction@12();
    VAR
      RFSetup@1000 : Record 14002801;
    BEGIN
      OWRAssistedSetup.SetStatus(PAGE::"Assisted OWR Setup Wizard",OWRAssistedSetup.Status::Completed);
      RFSetup.INIT;
      RFSetup := Rec;
      IF NOT RFSetup.INSERT(TRUE) THEN
        RFSetup.MODIFY(TRUE);
      CurrPage.CLOSE;
    END;

    LOCAL PROCEDURE NextStep@41(Backwards@1000 : Boolean);
    BEGIN
      IF Backwards THEN
        Step := Step - 1
      ELSE
        Step := Step + 1;

      EnableControls;
    END;

    LOCAL PROCEDURE ShowStartStep@24();

      BackActionEnabled := FALSE;
    END;

    LOCAL PROCEDURE ShowWizardFields@1240020002();
    VAR
      RFSetup@1240020002 : Record 14002801;
      OWRSingleInstance@1240020000 : Codeunit 14002864;
      IsExtension@1240020003 : Boolean;
      TestOverride@1240020001 : Boolean;
    BEGIN
      // This function will only apply to specific host versions
      TestOverride := OWRSingleInstance.GetTestCodeunitOverride;
      RFSetup.CheckIsExtension(IsExtension);

      IF TestOverride OR (NOT IsExtension) THEN BEGIN
        ShowWavePickFields := TRUE;
        ShowPerZoneFields := TRUE;
        ShowPickFromFilterFields := TRUE;
      END ELSE
        IF IsExtension THEN BEGIN
          ShowWavePickFields := TRUE;
          ShowPerZoneFields := TRUE;
          ShowPickFromFilterFields := TRUE;
        END;
    END;

    LOCAL PROCEDURE ResetControls@1();
    BEGIN
      FinishActionEnabled := FALSE;
      BackActionEnabled := TRUE;
      NextActionEnabled := TRUE;

      FirstStepVisible := FALSE;
      EnableStepVisible := FALSE;
      SettingsStepVisible := FALSE;
      FinalStepVisible := FALSE;
    END;

    LOCAL PROCEDURE LoadTopBanners@40();
    BEGIN
      IF MediaRepositoryStandard.GET('AssistedSetup-NoText-400px.png',FORMAT(ClientTypeManagement.GetCurrentClientType)) AND
         MediaRepositoryDone.GET('AssistedSetupDone-NoText-400px.png',FORMAT(ClientTypeManagement.GetCurrentClientType))
      THEN
        IF MediaResourcesStandard.GET(MediaRepositoryStandard."Media Resources Ref") AND
           MediaResourcesDone.GET(MediaRepositoryDone."Media Resources Ref")
        THEN
          TopBannerVisible := MediaResourcesDone."Media Reference".HASVALUE;
    END;

    BEGIN
    END.
