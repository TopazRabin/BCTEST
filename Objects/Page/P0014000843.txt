        SetTextFile;

        EXIT;
      END;

        Type::Barcode:
          SetBarcode;
        Type::Line:
          SetLine;
        Type::Text:
          SetText;
        Type::Zone:
          SetZone;
        Type::Bitmap:
          SetBitmap;
        Type::"2D Barcode":
          Set2DBarcode;
      END;
    END;
