                                  ItemUnitofMeasure2.RESET;

                                  ItemLedgerEntry.RESET;
                                  ItemLedgerEntry.SETRANGE("Item No.",Item."No.");
                                  ItemLedgerEntry.SETRANGE("Entry Type",ItemLedgerEntry."Entry Type"::Sale);
                                  ItemLedgerEntry.SETRANGE("Source No.",'50077');
                                  ItemLedgerEntry.SETRANGE("Document Type",ItemLedgerEntry."Document Type"::"Sales Shipment");
                                  //ItemLedgerEntry.SETRANGE("Posting Date",(CALCDATE('-CY',TODAY)),EndDate);         // Lowest price of item from 1st jan.
                                  ItemLedgerEntry.SETRANGE("Posting Date",StartDate,EndDate);
                                  IF ItemLedgerEntry.FINDSET THEN REPEAT
                                    PurchasedItem := TRUE;
                                    ItemLedgerEntry.CALCFIELDS("Sales Amount (Actual)","Sales Amount (Expected)");
                                    IF ItemLedgerEntry.Quantity * -1 > 0 THEN
                                      UnitPrice := ((ItemLedgerEntry."Sales Amount (Actual)" + ItemLedgerEntry."Sales Amount (Expected)") / (ItemLedgerEntry.Quantity * -1));
                                    IF LowestPrice = 0 THEN
                                      LowestPrice := UnitPrice
                                    ELSE IF LowestPrice > UnitPrice THEN
                                      LowestPrice := UnitPrice;
                                  UNTIL ItemLedgerEntry.NEXT = 0;
                                  IF PurchasedItem THEN BEGIN
                                    IF (Item."Manufacturer Code" = 'GE') OR (Item."Manufacturer Code" = 'OSRAM') OR (Item."Manufacturer Code" = 'PHILIPS') OR(Item."Manufacturer Code" = 'SYLVANIA') THEN BEGIN
                                      LowestPrice := (LowestPrice * 1.12);
                                    END;
                                  END;
                                  IF NOT PurchasedItem THEN BEGIN
                                    IF (Item."Manufacturer Code" = 'GE') OR (Item."Manufacturer Code" = 'OSRAM') OR (Item."Manufacturer Code" = 'PHILIPS') OR(Item."Manufacturer Code" = 'SYLVANIA') THEN
                                      LowestPrice := (Item."Medium Unit Price" * 1.12)
                                    ELSE
                                      LowestPrice := Item."Medium Unit Price";
                                  END;
                                END;
