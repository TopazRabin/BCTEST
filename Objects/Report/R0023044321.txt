    LOCAL PROCEDURE doCalcs@1000000027();
    VAR
      ldTotalInventoryAmt@1000000002 : Decimal;
      ldTotalInventoryQty@1000000001 : Decimal;
    BEGIN
      // Calculates inventory values and turns for current item in global recItem record

      recItem.SETRANGE("Date Filter", dtBeginDate, dtEndDate);
      recItem.CALCFIELDS(Inventory, "Sales (Qty.)", "Negative Adjmt. (Qty.)", "COGS (LCY)",
                         //"Negative Adjmt. (LCY)",
                                          "Transferred (Qty.)");

      //Converted code to not require consumptions (Lcy) and just use calcsums on the cost amount (Actual)
      recValueEntry.RESET();
      IF bIncludeProduction THEN BEGIN
        recItem.CALCFIELDS("Consumptions (Qty.)" );
        recValueEntry.SETRANGE("Item Ledger Entry Type", recValueEntry."Item Ledger Entry Type"::Consumption);
        recValueEntry.SETRANGE("Item No.",recItem."No.");
        recValueEntry.SETRANGE("Global Dimension 1 Code",recItem.GETFILTER("Global Dimension 1 Filter"));
        recValueEntry.SETRANGE("Global Dimension 2 Code",recItem.GETFILTER("Global Dimension 2 Filter"));
        recValueEntry.SETRANGE("Location Code",recItem.GETFILTER("Location Filter"));
        recValueEntry.SETFILTER("Drop Shipment",recItem.GETFILTER("Drop Shipment Filter"));
        recValueEntry.SETRANGE("Variant Code",recItem.GETFILTER("Variant Filter"));
        recValueEntry.SETFILTER("Posting Date",recItem.GETFILTER("Date Filter"));
        recValueEntry.CALCSUMS("Cost Amount (Actual)" );
        //
      END ELSE BEGIN
        recItem."Consumptions (Qty.)" := 0;
        //recItem."Consumptions (LCY)" := 0;
        recValueEntry."Cost Amount (Actual)" := 0;
      END;//if

      // get totals for period
      dClosingBalanceQty := recItem.Inventory;
      dClosingBalanceAmt := recItem.Inventory * recItem."Unit Cost";

      dMovementQty := recItem."Consumptions (Qty.)" + recItem."Sales (Qty.)" -
                      //recItem."Negative Adjmt. (Qty.)" +
                      recItem."Transferred (Qty.)";

      dMovementAmt := recItem."COGS (LCY)" - recValueEntry."Cost Amount (Actual)"
                      //- recItem."Negative Adjmt. (LCY)"
                      - (recItem."Transferred (Qty.)" * recItem."Unit Cost");

      // get average inventory


      getAverageInventory(dOpeningBalanceQty, dOpeningBalanceAmt);
      //Turns are no longer used, removed that portion of code
    END;

    LOCAL PROCEDURE getAverageInventory@1000000028(VAR pdOpeningBalanceQty@1000000002 : Decimal;VAR pdOpeningBalanceAmt@1000000006 : Decimal);
    VAR
      ldtDataPointDate@1000000004 : Date;
      lnNumDataPoints@1000000003 : Integer;
      ldTotalInventoryQty@1000000005 : Decimal;
      ldPrecision@1000000007 : Decimal;
    BEGIN
      // Get the average inventory calculated weekly

      ldtDataPointDate := dtBeginDate - 1;
      recItem.SETRANGE("Date Filter", 0D, ldtDataPointDate);
      recItem.CALCFIELDS("Net Change");

      ldTotalInventoryQty := recItem."Net Change";
      pdOpeningBalanceQty := recItem."Net Change";
      pdOpeningBalanceAmt := recItem."Net Change" * recItem."Unit Cost";

      // inventory turns not required for this report
      EXIT;
      // Removed code no longer needed, changed function signature.
    END;

    LOCAL PROCEDURE calcABCcode@1000000001();
    VAR
      lbFoundPeriod@1000000004 : Boolean;
      lbUpdateItem@1000000002 : Boolean;
      ldecPositionPct@1000000001 : Decimal;
      liLowerEndLimit@1000000000 : Integer;
      liUpperEndLimit@1000000003 : Integer;
      lrecItem@1000000005 : Record 27;
    BEGIN
      // Determines the item's ABC code, based on percentage algorithm
      codCountPeriod := trecInventoryBuffer."Phys Invt Counting Period Code"; // old value
      IF optCountCodeHandling <> optCountCodeHandling::"Do not calculate" THEN BEGIN
        ldecPositionPct := iOnRecNumber/iNoOfItemsToProcess*100;
        IF ldecPositionPct>100 THEN
          ldecPositionPct := 100;

        recCountPeriods.SETCURRENTKEY("Max. Pct. of Items by Cost");   // order records by increasing Thresholds
        recCountPeriods.SETFILTER("Max. Pct. of Items by Cost",'<>0'); // do not include records where the threshold was not set
        IF recCountPeriods.FIND('-') THEN BEGIN
          liLowerEndLimit := 0;
        END ELSE BEGIN
          ERROR(txtNoPhysInvtCountPeriodSet);
        END;

        REPEAT
          liUpperEndLimit := recCountPeriods."Max. Pct. of Items by Cost";
          IF (ldecPositionPct > liLowerEndLimit) AND (ldecPositionPct <= liUpperEndLimit) THEN BEGIN
            lbFoundPeriod := TRUE;
            codCountPeriod := recCountPeriods."Invt. Counting Period Code";
          END ELSE BEGIN
            liLowerEndLimit := liUpperEndLimit;
          END;
        UNTIL (recCountPeriods.NEXT=0) OR lbFoundPeriod;

        IF NOT (optCountCodeHandling = optCountCodeHandling::"Calculate only") THEN BEGIN
          lbUpdateItem := TRUE;
          IF optCountCodeHandling = optCountCodeHandling::"Calculate and update items/skus" THEN BEGIN
            recSKU.SETRANGE("Item No.", trecInventoryBuffer."Phys Invt Counting Period Code");
            IF recSKU.FINDSET(TRUE) THEN BEGIN
              lbUpdateItem := FALSE;
              REPEAT
                recSKU."Phys Invt Counting Period Code" := codCountPeriod;
                cuPhysCountMgmt.UpdateSKUPhysInvtCount(recSKU);
              UNTIL recSKU.NEXT = 0;
            END;//if got sku
          END;//if

          IF lbUpdateItem THEN BEGIN // for items update only or for items that don't have skus
            lrecItem.GET(trecInventoryBuffer."Item No.");
            lrecItem."Phys Invt Counting Period Code" := codCountPeriod;
            IF (lrecItem."Phys Invt Counting Period Code"<>'') THEN
              cuPhysCountMgmt.UpdateItemPhysInvtCount(lrecItem)
            ELSE
              lrecItem.MODIFY;
          END;//if
        END;
      END; // optCountCodeHandling <> optCountCodeHandling::"Do not calculate"

      // moved from top of function
      IF (optCountCodeHandling = optCountCodeHandling::"Calculate only") THEN
        codCountPeriod := '('+ codCountPeriod+')';
    END;

    PROCEDURE fillItem2TempRec@1000000002();
    BEGIN
      //<IW author="S.Roy" date="5/30/2011" issue="92" >
      iOnRecNumber += 1;
      IF TIME - tmTimeProgress > 2000 THEN BEGIN //every 2 seconds
        tmTimeProgress := TIME;
        iProgress := ROUND(iOnRecNumber/iNoOfItemsToProcess * 10000,1);
        dlgProgressWindow.UPDATE;
      END;

      trecInventoryBuffer.INIT;
      // Use IW Inventory buffer instead of Item 2
      //trecInventoryBuffer.TRANSFERFIELDS(recItem);
      trecInventoryBuffer."Item No." := recItem."No.";
      trecInventoryBuffer.Description := recItem.Description;
      trecInventoryBuffer."Description 2" := recItem."Description 2";
      trecInventoryBuffer."Phys Invt Counting Period Code" := recItem."Phys Invt Counting Period Code";
      trecInventoryBuffer."No. of Turns" := 0; //<IW author="t.d" date="1/12/2015" issue="TFS1616" />
      trecInventoryBuffer."Movement Quantity" := dMovementQty;
      trecInventoryBuffer."Opening Balance Qty" := dOpeningBalanceQty;
      trecInventoryBuffer."Closing Balance Qty" := dClosingBalanceQty;
      trecInventoryBuffer."Average Inventory Qty" := 0; //<IW author="t.d" date="1/12/2015" issue="TFS1616" />
      trecInventoryBuffer."Opening Balance Amount" := dOpeningBalanceAmt;
      trecInventoryBuffer."Closing Balance Amount" := dClosingBalanceAmt;
      trecInventoryBuffer."Average Inventory Amount" := 0;
      trecInventoryBuffer."Movement Amount" := dMovementAmt;
      trecInventoryBuffer."Calculated Count. Period Code" := codCountPeriod;
      trecInventoryBuffer.INSERT;
    END;

    PROCEDURE blankItemsSKUsCountPeriod@1000000000();
    VAR
      ltxtConfirmCountPeriodBlanking@1000000000 : TextConst 'ENU=Clear all Phys. Phys Invt Counting Period Codes on the %1?;ENC=Clear all Phys. Invt. Counting Period codes on the %1?';
      ltxtQuestionParameter@1000000001 : Text[30];
      lrecSKU@1000000002 : Record 5700;
      lrecItem@1000000003 : Record 27;
    BEGIN
      // By not clearing, it will make it more difficult to find items that
      // were not classified.
      IF ( NOT bClearCountingPeriodFirst ) THEN BEGIN
        EXIT;
      END;

      //<IW author="S.Roy" date="5/30/2011" issue="92" >
      ltxtQuestionParameter := ''; //make sure that variable is initialized properly
      IF (optCountCodeHandling = optCountCodeHandling::"Calculate and update items only") THEN BEGIN
        ltxtQuestionParameter := 'items';
      END;
      IF (optCountCodeHandling = optCountCodeHandling::"Calculate and update items/skus") THEN BEGIN
        ltxtQuestionParameter := 'items and skus';
      END;

      //only blank the appropriate fields if user set it up that way (controlled via ltxtQuestionParameter set above)
      IF ltxtQuestionParameter <> '' THEN BEGIN
        IF NOT CONFIRM(ltxtConfirmCountPeriodBlanking,TRUE,ltxtQuestionParameter) THEN BEGIN
          ERROR('');
        END ELSE BEGIN
          //if we get here, it's because items (and maybe skus) need to be reset
          IF (optCountCodeHandling = optCountCodeHandling::"Calculate and update items/skus") THEN BEGIN
            lrecSKU.SETFILTER("Phys Invt Counting Period Code",'<>%1','');

            //lrecSKU.MODIFYALL("Next Counting Period",'');
            //Previous behavior was to update the last counting period update as well.
            //lrecSKU.MODIFYALL("Last Counting Period Update",0D);
            lrecSKU.MODIFYALL("Phys Invt Counting Period Code",'');
          END;

          lrecItem.SETFILTER("Phys Invt Counting Period Code",'<>%1','');
          //lrecItem.MODIFYALL("Next Counting Period",'');
          //Previous behavior was to update the last counting period update as well.
          //lrecItem.MODIFYALL("Last Counting Period Update",0D);
          lrecItem.MODIFYALL("Phys Invt Counting Period Code",'');
        END; // of else NOT CONFIRM

      END;
    END;

    BEGIN
    END.
