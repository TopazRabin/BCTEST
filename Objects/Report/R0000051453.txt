                                    IF OnLineNumber = 1 THEN
                                      FIND('-')
                                    ELSE
                                      NEXT;

                                    // <TPZ197>
                                    IF (Type = Type::Item) OR
                                      ((Type = Type::"G/L Account") AND ("No." = SalesSetup."Freight G/L Acc. No."))
                                      THEN BEGIN
                                      LineNo := LineNo + 1;
                                      LineNoText := FORMAT(LineNo);
                                      DescriptionToPrint := Description;
                                    END ELSE BEGIN
                                      LineNoText := '';
                                      DescriptionToPrint:= Description + "Description 2";
                                    END;
                                    // </TPZ197>

                                    OrderedQuantity := 0;
                                    IF "Sales Invoice Header"."Order No." = '' THEN
                                      OrderedQuantity := Quantity
                                    ELSE
                                      IF OrderLine.GET(1,"Sales Invoice Header"."Order No.","Line No.") THEN
                                        OrderedQuantity := OrderLine.Quantity
                                      ELSE BEGIN
                                        ShipmentLine.SETRANGE("Order No.","Sales Invoice Header"."Order No.");
                                        ShipmentLine.SETRANGE("Order Line No.","Line No.");
                                        IF ShipmentLine.FIND('-') THEN
                                          REPEAT
                                            OrderedQuantity := OrderedQuantity + ShipmentLine.Quantity;
                                          UNTIL 0 = ShipmentLine.NEXT;
                                      END;

                                    DescriptionToPrint := Description + ' ' + "Description 2";
                                    IF Type = 0 THEN BEGIN
                                      IF OnLineNumber < NumberOfLines THEN BEGIN
                                        NEXT;
                                        IF Type = 0 THEN BEGIN
                                          DescriptionToPrint :=
                                            COPYSTR(DescriptionToPrint + ' ' + Description + ' ' + "Description 2",1,MAXSTRLEN(DescriptionToPrint));
                                          OnLineNumber := OnLineNumber + 1;
                                          SalesInvLine.NEXT;
                                        END ELSE
                                          NEXT(-1);
                                      END;
                                      "No." := '';
                                      "Unit of Measure" := '';
                                      Amount := 0;
                                      "Amount Including VAT" := 0;
                                      "Inv. Discount Amount" := 0;
                                      Quantity := 0;
                                    END; // ELSE
                                      //IF Type = Type::"G/L Account" THEN
                                      //  "No." := '';

                                  //********************
                                   AmountExclInvDisc := Amount + "Inv. Discount Amount";

                                  IF Type = Type::"G/L Account" THEN BEGIN
                                  //SC 10022017    IF "No." = SalesSetup."G/L Freight Account No." THEN
                                      IF "No." = SalesSetup."Freight G/L Acc. No." THEN
                                        FreightAmt := AmountExclInvDisc
                                      ELSE
                                        MiscChargeAmt := AmountExclInvDisc;
                                      //AmountExclInvDisc := 0;  //RS1.00
                                    END;

                                    IF (Type = Type::Resource) AND Resource.GET("No.") THEN BEGIN
                                      CASE Resource."Charge Type" OF
                                        Resource."Charge Type"::Freight:
                                          FreightAmt := AmountExclInvDisc;
                                        Resource."Charge Type"::Miscellaneous:
                                          MiscChargeAmt := AmountExclInvDisc;
                                      END;
                                      //AmountExclInvDisc := 0;
                                    END;

                                    //<TPZ2273>
                                    IF "Sales Invoice Header"."Shortcut Dimension 5 Code" = 'I' THEN BEGIN
                                      freightChargestoSkip := 0;
                                      SalesInvoiceLineFreight.RESET;
                                      SalesInvoiceLineFreight.SETCURRENTKEY("Document No.","Line No.");
                                      SalesInvoiceLineFreight.SETRANGE(SalesInvoiceLineFreight."Document No.",TempSalesInvoiceLine."Document No.");
                                      SalesInvoiceLineFreight.SETRANGE(SalesInvoiceLineFreight.Type,SalesInvoiceLineFreight.Type::"G/L Account");
                                      SalesInvoiceLineFreight.SETRANGE(SalesInvoiceLineFreight."No.",SalesSetup."Freight G/L Acc. No.");
                                      IF SalesInvoiceLineFreight.FINDSET THEN BEGIN
                                        REPEAT
                                          freightChargestoSkip := freightChargestoSkip + SalesInvoiceLineFreight.Amount + SalesInvoiceLineFreight."Inv. Discount Amount";
                                        UNTIL SalesInvoiceLineFreight.NEXT = 0;
                                      END;
                                      IF TempSalesInvoiceLine.Type = TempSalesInvoiceLine.Type :: "G/L Account" THEN BEGIN
                                       IF (TempSalesInvoiceLine."No." = SalesSetup."Freight G/L Acc. No.")  AND (freightChargestoSkip = 0)THEN
                                        CurrReport.SKIP;
                                      END;
                                    END;
                                    //</TPZ2273>

                                    IF Type = Type::"G/L Account" THEN
                                      "No." := '';

                                  //*****************
                                    IF "No." = '' THEN BEGIN
                                      HighDescriptionToPrint := DescriptionToPrint;
                                      LowDescriptionToPrint := '';
                                    END ELSE BEGIN
                                      HighDescriptionToPrint := '';
                                      LowDescriptionToPrint := DescriptionToPrint;
                                    END;

                                    IF Amount <> "Amount Including VAT" THEN
                                      TaxLiable := Amount
                                    ELSE
                                      TaxLiable := 0;

                                    //AmountExclInvDisc := Amount + "Inv. Discount Amount";

                                    //<TPZ1771>
                                   IF InvoiceDiscount>0 THEN
                                      InvoiceDiscounttxt:=txtdiscount;
                                   //</TPZ1771>

                                    IF Quantity = 0 THEN
                                      UnitPriceToPrint := 0 // so it won't print
                                    ELSE
                                      UnitPriceToPrint := ROUND(AmountExclInvDisc / Quantity,0.00001);

                                    IF (Type = Type::"G/L Account") OR (Type = Type::Resource) THEN
                                       AmountExclInvDisc:=0;
                                  END;
                                  IF OnLineNumber = NumberOfLines THEN
                                    PrintFooter := TRUE;
                                END;
                                 }
