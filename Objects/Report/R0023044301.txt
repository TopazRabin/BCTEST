    LOCAL PROCEDURE checkForMissingItems@1000000033();
    VAR
      lrecLocation@1000000000 : Record 14;
    BEGIN
      // Figures out which items+location+variant+bin are missing from the count sheet
      // and optionally adds them

      lrecLocation.SETFILTER(Code, codLocationFilter);

      IF lrecLocation.FINDSET THEN
        REPEAT // location loop
          evaluateLocation(lrecLocation);
        UNTIL lrecLocation.NEXT = 0;

      IF codLocationFilter = '' THEN BEGIN
        lrecLocation.INIT;  // clear code to test for blank location
        evaluateLocation(lrecLocation);
      END;//if
    END;

    LOCAL PROCEDURE evaluateLocation@1000000005(precLocation@1000000000 : Record 14);
    VAR
      lrecItemVariant@1000000001 : Record 5401;
    BEGIN
      recTempLine.RESET;
      recTempLine.COPYFILTERS(diInitSheetLine);
      recTempLine.SETRANGE("Item No.", recItem."No.");
      recTempLine.SETFILTER("Location Code", precLocation.Code);

      lrecItemVariant.SETRANGE("Item No.", recItem."No.");
      lrecItemVariant.SETFILTER(Code, codVariantFilter);
      IF lrecItemVariant.FINDSET THEN
       REPEAT // variant loop
         recTempLine.SETFILTER("Variant Code", lrecItemVariant.Code);
         evaluateVariant(precLocation, lrecItemVariant.Code);
       UNTIL lrecItemVariant.NEXT = 0;

      IF codVariantFilter='' THEN BEGIN
        recTempLine.SETRANGE("Variant Code");
        evaluateVariant(precLocation, '');
      END;//if
    END;

    LOCAL PROCEDURE evaluateVariant@1000000001(precLocation@1000000000 : Record 14;pcodVariantCode@1000000002 : Text);
    VAR
      lrecBinContent@1000000001 : Record 7302;
    BEGIN

      // make sure we check all bins if they are mandatory
      IF precLocation."Bin Mandatory" THEN BEGIN
        lrecBinContent.SETRANGE("Item No.", recItem."No.");
        lrecBinContent.SETRANGE("Location Code", precLocation.Code);
        IF pcodVariantCode<>'' THEN
          lrecBinContent.SETRANGE("Variant Code", pcodVariantCode);
        IF lrecBinContent.FINDSET THEN
          REPEAT
            recTempLine.SETRANGE("Bin Code", lrecBinContent."Bin Code");
            IF bIncludeInventoryOnly THEN BEGIN
              //lrecBinContent.SETFILTER("Date Filter", recItem."Date Filter");
              lrecBinContent.CALCFIELDS(Quantity);
            END;//if

            // only check missing items if bin qty<>0 or if we don't care about on-hand
            IF (NOT bIncludeInventoryOnly) OR (lrecBinContent.Quantity<>0) THEN
              testMissingItem(TRUE, precLocation.Code, pcodVariantCode, lrecBinContent."Bin Code");
          UNTIL lrecBinContent.NEXT = 0;
      END ELSE
        testMissingItem(FALSE, precLocation.Code, pcodVariantCode, '');
    END;

    LOCAL PROCEDURE testMissingItem@1000000036(pbBinMandatory@1000000000 : Boolean;pcodLocation@1000000001 : Code[20];pcodVariant@1000000002 : Text;pcodBin@1000000003 : Code[20]);
    BEGIN
      IF NOT recTempLine.FINDFIRST THEN BEGIN

        IF bIncludeInventoryOnly THEN BEGIN
          IF NOT pbBinMandatory THEN BEGIN
            recItemLedger.SETRANGE("Item No.", recItem."No.");
            recItemLedger.SETRANGE("Location Code", pcodLocation);
            recItemLedger.SETRANGE("Variant Code", pcodVariant);
            recItemLedger.SETRANGE("Posting Date", 0D, dtAsOfDate);
            recItemLedger.SETRANGE(Open, TRUE);
            IF NOT recItemLedger.FINDFIRST THEN
              EXIT;
          END;//if inventory only
        END;

        IF txtComment = '' THEN
          txtComment := tcNotOnSheet;

        txtMarker := '*';

        //<IW author="R.T" date="2/8/2012" issue="296" >
        IF( (codAddToSheet<>'') OR (codAddToCountNo<>'')) THEN BEGIN
          cuInvMgmt.insertCountSheetLine(
            recItem,
            codAddToCountNo,
            codAddToSheet,
            pcodLocation,
            pcodVariant,
            '',
            pcodBin);
          txtComment := tcAddedToSheet;
        END;//if

      END;//no count line found
    END;
