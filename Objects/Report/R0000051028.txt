    PROCEDURE NoOfLinesInvoice@1000000000(SalesInvoiceHeader@1000000000 : Record 112) : Integer;
    VAR
      SalesInvoiceLine@1000000001 : Record 113;
    BEGIN
      //<TPZ1911>
      SalesInvoiceLine.RESET;

    PROCEDURE CostPerOrder1@1000000003(SalesHeader@1000000000 : Record 36) : Decimal;
    VAR
      SalesLine@1000000001 : Record 37;
      OrderCost@1000000002 : Decimal;
    BEGIN
      //<TPZ1911>
      OrderCost := 0;
      SalesLine.RESET;
      SalesLine.SETRANGE("Document No.",SalesHeader."No.");
      SalesLine.SETFILTER(Quantity,'<>%1',0);
      IF SalesLine.FINDSET THEN BEGIN
        REPEAT
          OrderCost := OrderCost + (SalesLine."Unit Cost (LCY)" * SalesLine.Quantity);
        UNTIL SalesLine.NEXT = 0;
        EXIT (OrderCost);
      END ELSE
        EXIT(0);
      //</TPZ1911>
    END;
