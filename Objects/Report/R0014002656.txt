    LOCAL PROCEDURE CalcUsage@1240020000(VAR CurrProcUnit@1240020000 : Record 14000555) AvgUsg : Decimal;
    VAR
      ItemUsage@1240020001 : Record 14000557;
      i@1240020002 : Integer;
      PeriodsWithUsg@1240020004 : Decimal;
    BEGIN
      AvgUsg := 0;

      ItemUsage.SETRANGE("Location Code", CurrProcUnit."Location Code");
      ItemUsage.SETRANGE("Item No.",CurrProcUnit."Item No.");
      ItemUsage.SETRANGE("Variant Code",CurrProcUnit."Variant Code");
      ItemUsage.SETFILTER("Source No.",'%1','');
      ItemUsage.SETFILTER("Starting Date",'%1..%2', HistBegin_StartingDate, HistEnd_StartingDate);
      IF ItemUsage.FINDSET THEN BEGIN
        i := 0;
        REPEAT
          i := i + 1;
          IF (ItemUsage."Net Period End Usage" > 0) THEN BEGIN
            AvgUsg := AvgUsg + ItemUsage."Net Period End Usage";
