    PROCEDURE GetCountry@1(CountryCode@1240030000 : Code[10]);
    BEGIN

        CLEAR(Country);

    PROCEDURE GetPackageNo@1240030001(PackageNo@1240030000 : Code[20]) : Code[20];

      i@1240030001 : Integer;
    BEGIN
      i := STRPOS(PackageNo,'-');
      IF i > 0 THEN
        EXIT(COPYSTR(PackageNo,1,i-1))
      ELSE
        EXIT(PackageNo);
    END;

    END;
