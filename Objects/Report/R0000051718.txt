    LOCAL PROCEDURE CheckAgainstPrevLines@20(CommJnlLine@1000 : Record 51705);
    VAR
      i@1001 : Integer;
      AccType@1002 : Integer;
      AccNo@1003 : Code[20];
      ErrorFound@1004 : Boolean;
    BEGIN
      AccType := CommJnlLine."Account Type";
      AccNo := CommJnlLine."Account No.";

      TempCommJnlLine.RESET;
      TempCommJnlLine.SETRANGE("Document No.",CommJnlLine."Document No.");

      WHILE (i < 2) AND NOT ErrorFound DO BEGIN
        i := i + 1;
        IF i = 1 THEN BEGIN
          TempCommJnlLine.SETRANGE("Account Type",AccType);
          TempCommJnlLine.SETFILTER("Account No.",'<>%1',AccNo);
          IF TempCommJnlLine.FINDFIRST THEN BEGIN
            ErrorFound := TRUE;
            AddError(
              STRSUBSTNO(
                Text064,CommJnlLine.FIELDCAPTION("Document No."),CommJnlLine."Document No.",
                TempCommJnlLine."Line No.",CommJnlLine.FIELDCAPTION("Document No."),TempCommJnlLine."Document No."));
          END;
        END;
      END;

      TempCommJnlLine.RESET;
      TempCommJnlLine := CommJnlLine;
      TempCommJnlLine.INSERT;
    END;
