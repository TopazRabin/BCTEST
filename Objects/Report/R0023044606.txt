                                  lrecItem@1000000001 : Record 27;
                                BEGIN

                                  codCountryOfOrigin := '';
                                  codTariffNo := '';
                                  dItemValue := 0;
                                  dItemGrossWeight := 0;

                                  // If the customs card values were overwritten, we want to use those
                                  IF ( lrecCustomsLine.GET(codCustomsHeader, diLPLine."Item No.", diLPLine."Variant Code") ) THEN BEGIN
                                    codCountryOfOrigin := lrecCustomsLine."Origin Country/Region Code";
                                    codTariffNo := lrecCustomsLine."HS Tariff No.";
                                    dItemValue := lrecCustomsLine."Value Per Item Qty.";
                                    dItemGrossWeight := lrecCustomsLine."Weight Per Item Qty.";
                                  END;

                                  // Don't check item if customs card had all the info
                                  IF ( (codCountryOfOrigin = '') OR
                                       (codTariffNo = '') OR
                                       (dItemValue = 0) OR
                                       (dItemGrossWeight = 0) ) THEN BEGIN

                                    lrecItem.GET(diLPLine."Item No.");

                                    IF ( codCountryOfOrigin = '' ) THEN BEGIN
                                      // if no country of origin, use other global for company country
                                      IF ( lrecItem."Country/Region of Origin Code" <> '' ) THEN
                                        codCountryOfOrigin := lrecItem."Country/Region of Origin Code"
                                      ELSE
                                        codCountryOfOrigin := codCountryOfCompany;
                                    END;

                                    IF ( codTariffNo = '' ) THEN BEGIN
                                      codTariffNo := lrecItem."Tariff No.";
                                    END;

                                    IF ( dItemValue = 0 ) THEN BEGIN
                                      IF ( bIsSalesOrder ) THEN
                                        dItemValue := lrecItem."Unit Price"
                                      ELSE
                                        dItemValue := lrecItem."Unit Cost";
                                    END;

                                    IF ( dItemGrossWeight = 0 ) THEN BEGIN
                                      dItemGrossWeight := lrecItem."Gross Weight";
                                    END;
                                  END;
                                END;

    LOCAL PROCEDURE getDetailsFromLP@1000000007(VAR precLPHeader@1000000000 : Record 23044505);
    VAR
      lrecWhseShipHeader@1000000002 : Record 7320;
      lrecWhseShipLine@1000000007 : Record 7321;
      lrecSalesHeader@1000000003 : Record 36;
      lrecTransferHeader@1000000004 : Record 5740;
      lcuListMgmt@1000000005 : Codeunit 23044600;
      lcuRateMgmt@1000000006 : Codeunit 23044602;
      codLocation@1000000001 : Code[20];
    BEGIN

      // get address details
      CASE precLPHeader."Source Document" OF
        precLPHeader."Source Document"::Shipment:
          BEGIN
            IF ( lrecWhseShipHeader.GET(precLPHeader."Source No.") ) THEN BEGIN
              codLocation := lrecWhseShipHeader."Location Code";
              lcuRateMgmt.setShipToAddressForWhseShip(lrecWhseShipHeader, TRUE, asShipToDetails);
              // check if whse ship is sales order
              lrecWhseShipLine.SETRANGE("No.", lrecWhseShipHeader."No.");
              IF ( lrecWhseShipLine.FINDFIRST ) THEN
                bIsSalesOrder := lrecWhseShipLine."Source Document" = lrecWhseShipLine."Source Document"::"Sales Order";
            END;
          END;
        precLPHeader."Source Document"::"Sales Order":
          BEGIN
            IF ( lrecSalesHeader.GET(lrecSalesHeader."Document Type"::Order, precLPHeader."Source No.") ) THEN BEGIN
              codLocation := lrecSalesHeader."Location Code";
              lcuRateMgmt.setShipToAddressForSalesOrder(lrecSalesHeader, TRUE, asShipToDetails);
              bIsSalesOrder := TRUE;
            END;
          END;
        precLPHeader."Source Document"::"Outbound Transfer":
          BEGIN
            IF ( lrecTransferHeader.GET(precLPHeader."Source No.") ) THEN BEGIN
              codLocation := lrecTransferHeader."Transfer-from Code";
              lcuRateMgmt.getShipToAddressFromTransferHeader(lrecTransferHeader, TRUE, asShipToDetails);
              bIsSalesOrder := FALSE;
            END;
          END;
      END;

      lcuRateMgmt.getShipFromAddress(codLocation, asShipFromDetails);

      // set tax id
      IF ( asShipFromDetails[11] <> '' ) THEN BEGIN
        sShipFromTaxID := asShipFromDetails[11];
      END ELSE IF ( asShipFromDetails[12] <> '' ) THEN BEGIN
        sShipFromTaxID := asShipFromDetails[12];
      END;

      IF ( asShipToDetails[11] <> '' ) THEN BEGIN
        sShipToTaxID := asShipToDetails[11];
      END ELSE IF ( asShipToDetails[12] <> '' ) THEN BEGIN
        sShipToTaxID := asShipToDetails[12];
      END;

      // processed order, prevent subsequent runs
      bGotOrder := TRUE;
    END;

    LOCAL PROCEDURE getDetailsFromPostedLP@1000000008(VAR precLPHeader@1000000000 : Record 23044505);
    VAR
      lrecPostedWhseShipHeader@1000000008 : Record 7322;
      lrecPostedWhseShipLine@1000000009 : Record 7323;
      lrecSalesShipmentHeader@1000000010 : Record 110;
      lrecTransferShipmentHeader@1000000011 : Record 5744;
      lcuListMgmt@1000000005 : Codeunit 23044600;
      lcuRateMgmt@1000000006 : Codeunit 23044602;
      codLocation@1000000001 : Code[20];
    BEGIN
      //<FUNC>
      //  Description: Gets address details from posted LP and docs
      //  Called From: This Report (R23044606)
      //  Side Effect: None
      //</FUNC>

      // get address details
      CASE precLPHeader."Shipped Source Document" OF
        precLPHeader."Shipped Source Document"::Shipment:
          BEGIN
            IF ( lrecPostedWhseShipHeader.GET(precLPHeader."Shipment No.") ) THEN BEGIN
              codLocation := lrecPostedWhseShipHeader."Location Code";
              lrecPostedWhseShipLine.SETRANGE("No.", lrecPostedWhseShipHeader."No.");
              IF ( lrecPostedWhseShipLine.FINDFIRST ) THEN
                bIsSalesOrder := lrecPostedWhseShipLine."Source Document" = lrecPostedWhseShipLine."Source Document"::"Sales Order";
              lcuRateMgmt.setShipToAddressForPostedWhseShip(lrecPostedWhseShipHeader, TRUE, asShipToDetails);
            END;
          END;
        precLPHeader."Shipped Source Document"::"Sales Order":
          BEGIN
            IF ( lrecSalesShipmentHeader.GET(precLPHeader."Shipment No.") ) THEN BEGIN
              codLocation := lrecSalesShipmentHeader."Location Code";
              bIsSalesOrder := TRUE;
              lcuRateMgmt.getShipToAddressFromSalesShipmentHeader(lrecSalesShipmentHeader, TRUE, asShipToDetails);
            END;
          END;
        precLPHeader."Shipped Source Document"::"Outbound Transfer":
          BEGIN
            IF ( lrecTransferShipmentHeader.GET(precLPHeader."Shipment No.") ) THEN BEGIN
              codLocation := lrecTransferShipmentHeader."Transfer-from Code";
              bIsSalesOrder := FALSE;
              lcuRateMgmt.getShipToAddressFromTransferShipmentHeader(lrecTransferShipmentHeader, TRUE, asShipToDetails);
            END;
          END;
      END;

      lcuRateMgmt.getShipFromAddress(codLocation, asShipFromDetails);

      // set tax id
      IF ( asShipFromDetails[11] <> '' ) THEN BEGIN
        sShipFromTaxID := asShipFromDetails[11];
      END ELSE IF ( asShipFromDetails[12] <> '' ) THEN BEGIN
        sShipFromTaxID := asShipFromDetails[12];
      END;

      IF ( asShipToDetails[11] <> '' ) THEN BEGIN
        sShipToTaxID := asShipToDetails[11];
      END ELSE IF ( asShipToDetails[12] <> '' ) THEN BEGIN
        sShipToTaxID := asShipToDetails[12];
      END;

      // processed order, prevent subsequent runs
      bGotOrder := TRUE;
    END;
