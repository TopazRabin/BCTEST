    LOCAL PROCEDURE setupReport@1000000009();
    VAR
      ltrecCountLineBins@1000000000 : TEMPORARY Record 23044301;
      ltcErrNoCountSheet@1000000001 : TextConst 'ENU=No recount sheet name has been specified.';
    BEGIN
      // Initializes report records, etc.

      IF codRecountCountNo='' THEN
        codRecountCountNo := diInitSheetLine.GETFILTER("Count No.");

      IF bAddToSheet AND ((codRecountSheet='') OR (codRecountCountNo='')) THEN
        ERROR(ltcErrNoCountSheet);

      // roll up inventory regardless of bin
      cuInvCountMgmt.getInventoryDifference(diInitSheetLine, ltrecCountLineBins,
        FALSE,
        TRUE, // group by lot
        FALSE // Group by UOM //<IW author="MH" date="7/22/2015" issue="TFS1895" />
        );

      cuInvCountMgmt.rollUpBinLines(ltrecCountLineBins, trecCountSummary);

      ltrecCountLineBins.RESET;
      ltrecCountLineBins.DELETEALL;

      trecCountSummary.RESET;


      // do amounts
      IF dAmountThreshold<>0 THEN BEGIN
        trecCountSummary.SETFILTER("Difference Amount (ABS)", '>=%1', dAmountThreshold);
        insertRecountDetails;
      END;//if

      // do quantities
      IF dQtyThreshold<>0 THEN BEGIN
        trecCountSummary.RESET;
        trecCountSummary.SETFILTER("Difference Qty. (ABS) (Base)", '>=%1', dQtyThreshold);
        insertRecountDetails;
      END;//if qty

      // do % difference
      IF( dPercentThreshold <> 0 ) THEN BEGIN
        trecCountSummary.RESET;
        trecCountSummary.SETFILTER("Difference Qty. Pct. (ABS)", '>=%1', dPercentThreshold);
        insertRecountDetails;
      END;

      // do all
      IF (dQtyThreshold=0) AND (dAmountThreshold=0) AND (dPercentThreshold=0) THEN BEGIN //<IW author="RT" date="6/08/15" issue="TFS1853" />
        trecCountSummary.RESET;
        insertRecountDetails;
      END;//if
    END;

    PROCEDURE insertRecountDetails@1000000015();
    VAR
      lrecCountSheetLine@1000000000 : Record 23044301;
    BEGIN
      // Adds details for recount to the temp record for printing and final processing

      IF trecCountSummary.FINDSET THEN BEGIN
        REPEAT
          lrecCountSheetLine.COPYFILTERS(diInitSheetLine);
          lrecCountSheetLine.SETRANGE("Item No.", trecCountSummary."Item No.");
          lrecCountSheetLine.SETRANGE("Location Code", trecCountSummary."Location Code");
          lrecCountSheetLine.SETRANGE("Recount Tag No.", 0);
          lrecCountSheetLine.SETRANGE("WHI Kit Item", FALSE);

          //<IW author="Howie Zhao" date="13/04/2015" issue="TFS1716" >
          IF (optCountByLineOrItem = optCountByLineOrItem::"Count Line") THEN BEGIN
            IF dAmountThreshold<>0 THEN BEGIN
              lrecCountSheetLine.SETFILTER("Difference Amount (ABS)", '>=%1', dAmountThreshold);
            END;
            IF dQtyThreshold<>0 THEN BEGIN
              lrecCountSheetLine.SETFILTER("Difference Qty. (ABS) (Base)", '>=%1', dQtyThreshold);
            END;//if qty

            IF( dPercentThreshold <> 0 ) THEN BEGIN
              lrecCountSheetLine.SETFILTER("Difference Qty. Pct. (ABS)", '>=%1', dPercentThreshold);
            END; // % difference

          END ELSE BEGIN
            // do nothing, because threshold by item was already done by this point (classic behavior).
          END;

          IF bIgnoreUncounted THEN
            lrecCountSheetLine.SETRANGE(Counted, TRUE);

          IF lrecCountSheetLine.FINDSET THEN BEGIN
            REPEAT
              IF lrecCountSheetLine."Recount Tag No."=0 THEN BEGIN
              trecRecountLines := lrecCountSheetLine;
              IF trecRecountLines.INSERT THEN ;
              END;//if
            UNTIL lrecCountSheetLine.NEXT = 0;
          END;//if found detail
        UNTIL trecCountSummary.NEXT = 0;
      END;//if
    END;
