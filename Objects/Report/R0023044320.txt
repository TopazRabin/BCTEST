    LOCAL PROCEDURE doCalcs@1000000027();
    VAR
      ldTotalInventoryAmt@1000000002 : Decimal;
      ldTotalInventoryQty@1000000001 : Decimal;
    BEGIN
      // Calculates inventory values and turns for current item in global recItem record

      recItem.SETRANGE("Date Filter", dtBeginDate, dtEndDate);
      recItem.CALCFIELDS(Inventory, "Sales (Qty.)", "Negative Adjmt. (Qty.)", "COGS (LCY)",
                         "Negative Adjmt. (LCY)",
                          "Transferred (Qty.)"); // NAV 2009
                         // "Transfer Shipments (Qty.)"); // NAV 5

      //Converted code to not require consumptions (Lcy) and just use calcsums on the cost amount (Actual)
      recValueEntry.RESET();
      IF bIncludeProduction THEN BEGIN
        recValueEntry.SETRANGE("Item Ledger Entry Type", recValueEntry."Item Ledger Entry Type"::Consumption);
        recValueEntry.SETRANGE("Item No.",recItem."No.");
        recValueEntry.SETRANGE("Global Dimension 1 Code",recItem.GETFILTER("Global Dimension 1 Filter"));
        recValueEntry.SETRANGE("Global Dimension 2 Code",recItem.GETFILTER("Global Dimension 2 Filter"));
        recValueEntry.SETRANGE("Location Code",recItem.GETFILTER("Location Filter"));
        recValueEntry.SETFILTER("Drop Shipment",recItem.GETFILTER("Drop Shipment Filter"));
        recValueEntry.SETRANGE("Variant Code",recItem.GETFILTER("Variant Filter"));
        recValueEntry.SETFILTER("Posting Date",recItem.GETFILTER("Date Filter"));
        recItem.CALCFIELDS("Consumptions (Qty.)");
        recValueEntry.CALCSUMS("Cost Amount (Actual)");
      END ELSE BEGIN
        recItem."Consumptions (Qty.)" := 0;
        recValueEntry."Cost Amount (Actual)" := 0;
      END;//if

      // get totals for period
      dClosingBalanceQty := recItem.Inventory;
      dClosingBalanceAmt := recItem.Inventory * recItem."Unit Cost";

      dMovementQty := recItem."Consumptions (Qty.)" + recItem."Sales (Qty.)" +
                      recItem."Negative Adjmt. (Qty.)" +
                      recItem."Transferred (Qty.)"; // NAV 2009
                      // recItem."Transfer Shipments (Qty.)"; // NAV 5

      dMovementAmt := recItem."COGS (LCY)" - recValueEntry."Cost Amount (Actual)" -
                      recItem."Negative Adjmt. (LCY)" +
                      recItem."Transferred (Qty.)"  // NAV 2--0
                      // recItem."Transfer Shipments (Qty.)"  // NAV 5
                      * recItem."Unit Cost";



      // get average inventory
      getAverageInventory(dAvgInventoryQty, dAvgInventoryAmt, dOpeningBalanceQty, dOpeningBalanceAmt);

      // calc # of turns and movement $
      IF (dAvgInventoryQty <> 0) THEN
        dNoOfTurns := dMovementQty / dAvgInventoryQty
      ELSE
        dNoOfTurns := 0;

      IF dNoOfTurns <0 THEN
        dNoOfTurns := 0;

      IF (dNoOfTurns < dMinTurns) OR ((dMaxTurns<>0) AND (dNoOfTurns > dMaxTurns)) THEN
        CurrReport.SKIP;

      calcCountPeriod;
    END;

    LOCAL PROCEDURE getAverageInventory@1000000028(VAR pdAvgInventoryQty@1000000000 : Decimal;VAR pdAvgInventoryAmt@1000000001 : Decimal;VAR pdOpeningBalanceQty@1000000002 : Decimal;VAR pdOpeningBalanceAmt@1000000006 : Decimal);
    VAR
      ldtDataPointDate@1000000004 : Date;
      lnNumDataPoints@1000000003 : Integer;
      ldTotalInventoryQty@1000000005 : Decimal;
      ldPrecision@1000000007 : Decimal;
    BEGIN
      // Get the average inventory calculated weekly

      ldtDataPointDate := dtBeginDate - 1;
      recItem.SETRANGE("Date Filter", 0D, ldtDataPointDate);
      recItem.CALCFIELDS("Net Change");

      ldTotalInventoryQty := recItem."Net Change";
      pdOpeningBalanceQty := recItem."Net Change";
      pdOpeningBalanceAmt := recItem."Net Change" * recItem."Unit Cost";

      lnNumDataPoints := 1;

      // get the total inventory amount summed by period to use with average calc
      REPEAT
        ldtDataPointDate := CALCDATE(TimeBetweenDataPoints, ldtDataPointDate);
        IF ldtDataPointDate > dtEndDate THEN
          ldtDataPointDate := dtEndDate;

        recItem.SETRANGE("Date Filter", 0D, ldtDataPointDate);
        recItem.CALCFIELDS("Net Change");
        ldTotalInventoryQty += recItem."Net Change";

        lnNumDataPoints += 1;
      UNTIL ldtDataPointDate = dtEndDate;

      ldPrecision := recItem."Rounding Precision";
      IF ldPrecision = 0 THEN ldPrecision := 1;
      IF bRoundUp THEN BEGIN
        pdAvgInventoryQty := ROUND(ldTotalInventoryQty / lnNumDataPoints, ldPrecision, '>');
      END ELSE BEGIN
        pdAvgInventoryQty := ROUND(ldTotalInventoryQty / lnNumDataPoints, ldPrecision);
      END;

      pdAvgInventoryAmt := recItem."Unit Cost" * pdAvgInventoryQty;
    END;

    LOCAL PROCEDURE calcCountPeriod@1000000001();
    VAR
      lbFound@1000000000 : Boolean;
      lbUpdateItem@1000000002 : Boolean;
    BEGIN
      // Determines the count code based on data in physical inventory count code table

      codCountPeriod := '('+ recItem."Phys Invt Counting Period Code"+')';

      IF bCalcCountCode AND (recCountPeriods.FIND('-')) THEN BEGIN
        REPEAT
          IF dNoOfTurns>=recCountPeriods."Min. Number of Turns" THEN BEGIN
            // got the right count period, so set it
            codCountPeriod := recCountPeriods."Invt. Counting Period Code";
            lbUpdateItem := TRUE;

            IF bUpdateSKUs THEN BEGIN
              recSKU.SETRANGE("Item No.", recItem."No.");
              IF recSKU.FINDSET THEN BEGIN
                lbUpdateItem := FALSE;
                recSKU.MODIFYALL("Phys Invt Counting Period Code", recCountPeriods."Invt. Counting Period Code");
              END;//if got SKU
            END;

            IF lbUpdateItem THEN BEGIN
              recItem."Phys Invt Counting Period Code" := recCountPeriods."Invt. Counting Period Code";
              recItem.MODIFY;
            END;//if

            lbFound := TRUE;
          END;//if
        UNTIL (recCountPeriods.NEXT=0) OR lbFound;
      END;//if
    END;
