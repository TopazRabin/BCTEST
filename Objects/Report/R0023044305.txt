    LOCAL PROCEDURE calculatePageCount@1000000016();
    VAR
      lrecProdOrderLine@1000000000 : Record 5406;
      lrecCountLines@1000000001 : Record 23044301;
      lnItems@1000000002 : Integer;
    BEGIN
      // Emulates report loops to calculate total page count

      cnItemsPerPage := 26; // NOTE: Change this if report layout changes

      lrecProdOrderLine.SETRANGE(Status, lrecProdOrderLine.Status::Released);
      lrecProdOrderLine.SETFILTER("Remaining Quantity", '<>%1', 0);
      lrecProdOrderLine.COPYFILTERS(recProdOrderLine);
      IF lrecProdOrderLine.FIND('-') THEN ;
      nTotalPages := 0;

      REPEAT
        lrecCountLines.SETFILTER("Prod. Order No.", '=%1', recProdOrderLine."Prod. Order No.");
        IF lrecCountLines.FIND('-') THEN
          nTotalPages += ROUND(lrecCountLines.COUNT/cnItemsPerPage, 1, '>');
      UNTIL lrecProdOrderLine.NEXT = 0;
    END;

    PROCEDURE getExpectedQty@1000000006() : Decimal;
    VAR
      dExpQty@1000000000 : Decimal;
    BEGIN
      // returns total expected qty of component on work order

      recProdOrderComp.SETFILTER("Prod. Order No.", '%1', recProdOrderLine."Prod. Order No.");
      recProdOrderComp.SETFILTER("Item No.", '%1', recCountSheetLine."Item No.");

      IF recProdOrderComp.FIND('-') THEN
        REPEAT
          dExpQty += recProdOrderComp."Expected Quantity";
        UNTIL recProdOrderComp.NEXT = 0;

      EXIT(dExpQty);
    END;
