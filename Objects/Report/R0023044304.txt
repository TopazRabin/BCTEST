    LOCAL PROCEDURE calculatePageCount@1000000016();
    VAR
      lrecWorkCenter@1000000000 : Record 99000754;
      lrecCountLines@1000000001 : Record 23044301;
      lnItems@1000000002 : Integer;
    BEGIN
      // Emulates report loops to calculate total page count

      cnItemsPerPage := 26; // NOTE: Change this if report layout changes

      lrecWorkCenter.COPYFILTERS(recWorkCenter);
      lrecWorkCenter.FIND('-');
      nTotalPages := 0;

      REPEAT
        trecActiveRtngLine.SETRANGE("Work Center No.", lrecWorkCenter."No.");
        IF trecActiveRtngLine.FIND('-') THEN BEGIN
          nTotalPages += 1;  // for blank sheet after each W/C
          lnItems := 0;
          REPEAT
            lrecCountLines.SETRANGE("Prod. Order No.", trecActiveRtngLine."Prod. Order No.");
            lnItems += lrecCountLines.COUNT;
          UNTIL trecActiveRtngLine.NEXT = 0;
          nTotalPages += ROUND(lnItems/cnItemsPerPage, 1, '>');
        END; //if
      UNTIL lrecWorkCenter.NEXT = 0;
    END;
