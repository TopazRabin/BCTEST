    PROCEDURE CreateTempPackageLineDetail@1000000001(VAR CurrPackageLine@1000000000 : Record 14000702);
    VAR
      QtyPacked@1000000003 : Decimal;
      QtyNeedToPack@1000000002 : Decimal;
      PackageLine3@1000000001 : Record 14000702;
      SalesLineLocal@1000000004 : Record 37;
    BEGIN
      TempPackageLineDetail.RESET;
      TempPackageLineDetail.DELETEALL;

      IF CurrPackageLine."Source Type" = DATABASE::"Transfer Header" THEN BEGIN
        TempPackageLineDetail.INIT;
        TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
        TempPackageLineDetail.INSERT;
        EXIT;
      END;

      IF CurrPackageLine."Source Type" = DATABASE::"Purchase Header" THEN BEGIN
        TempPackageLineDetail.INIT;
        TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
        TempPackageLineDetail.INSERT;
        EXIT;
      END;

      WITH CurrPackageLine DO BEGIN
        IF ("Source Type" = DATABASE::"Sales Header") AND (Type = Type::Item)THEN BEGIN
          PackageLine3.RESET;
          PackageLine3.SETFILTER("Package No.", '<%1', "Package No.");
          PackageLine3.SETRANGE("Source Type", "Source Type");
          PackageLine3.SETRANGE("Source Subtype", "Source Subtype");
          PackageLine3.SETRANGE("Source ID", "Source ID");
          PackageLine3.SETRANGE("No.", "No.");

          PackageLine3.SETRANGE("Location Code", "Location Code");

          CreateTempSalesLine(CurrPackageLine, PackageLine3);

          PackageLine3.RESET;
          PackageLine3.SETRANGE("Package No.", "Package No.");
          PackageLine3.SETRANGE("Source Type", "Source Type");
          PackageLine3.SETRANGE("Source Subtype", "Source Subtype");
          PackageLine3.SETRANGE("Source ID", "Source ID");
          PackageLine3.SETRANGE("No.", "No.");

          PackageLine3.SETRANGE("Location Code", "Location Code");

          PackageLine3.SETFILTER("Line No.", '<%1', "Line No.");
          CreateTempSalesLine(CurrPackageLine, PackageLine3);

          //////////////////////////////////////////////////////////////////////
          QtyPacked := Quantity;
          SalesLineLocal.RESET;
          SalesLineLocal.SETRANGE("Document Type", "Source Subtype");
          SalesLineLocal.SETRANGE("Document No.", "Source ID");
          SalesLineLocal.SETRANGE("No.", "No.");
          //SalesLineLocal.SETFILTER("Qty. to Ship", '>0');
          SalesLineLocal.SETRANGE("Location Code", "Location Code");

          IF SalesLineLocal.FIND('-') THEN
          REPEAT
            IF TempSalesLine.GET(SalesLineLocal."Document Type", SalesLineLocal."Document No.",
                                 SalesLineLocal."Line No.") = FALSE THEN BEGIN
               TempSalesLine.INIT;
               TempSalesLine.TRANSFERFIELDS(SalesLineLocal);
               TempSalesLine."Qty. to Ship" := 0;
               TempSalesLine."Qty. to Ship (Base)" := 0;
               TempSalesLine.INSERT;
            END;
            QtyNeedToPack := SalesLineLocal."Outstanding Quantity" - TempSalesLine."Qty. to Ship";

            IF QtyNeedToPack > 0 THEN BEGIN
              TempPackageLineNo := TempPackageLineNo + 10;
              TempPackageLineDetail.INIT;
              TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
              TempPackageLineDetail."Line No." := TempPackageLineNo;

              IF QtyNeedToPack >= QtyPacked THEN BEGIN
                TempPackageLineDetail.Quantity := QtyPacked;
                QtyPacked := 0;
              END ELSE BEGIN
                TempPackageLineDetail.Quantity := QtyNeedToPack;
                QtyPacked := QtyPacked - QtyNeedToPack;
              END;

              TempPackageLineDetail."Quantity (Base)" := TempPackageLineDetail.Quantity *
                                                            TempPackageLineDetail."Qty. per Unit of Measure";
              TempPackageLineDetail."Source Line No." := SalesLineLocal."Line No.";
              TempPackageLineDetail.INSERT;

              TempSalesLine."Qty. to Ship" := TempSalesLine."Qty. to Ship" +
                                                        TempPackageLineDetail.Quantity;
              TempSalesLine."Qty. to Ship (Base)" := TempSalesLine."Qty. to Ship (Base)" +
                                                        TempPackageLineDetail."Quantity (Base)";
              TempSalesLine.MODIFY;

            END;
          UNTIL (SalesLineLocal.NEXT = 0) OR (QtyPacked <= 0);

          IF QtyPacked > 0 THEN
            MESSAGE(Text00001, "No.");

        END;
      END;
    END;

    PROCEDURE CreateTempSalesLine@1000000000(VAR CurrPackageLine@1000000003 : Record 14000702;VAR PackageLine3@1000000000 : Record 14000702);
    VAR
      QtyPacked@1000000002 : Decimal;
      QtyNeedToPack@1000000001 : Decimal;
      SalesLineLocal@1000000004 : Record 37;

      WITH CurrPackageLine DO
      BEGIN
        IF PackageLine3.FIND('-') THEN
        REPEAT
          QtyPacked := PackageLine3.Quantity;
          SalesLineLocal.RESET;
          SalesLineLocal.SETRANGE("Document Type", "Source Subtype");
          SalesLineLocal.SETRANGE("Document No.", "Source ID");
          SalesLineLocal.SETRANGE("No.", "No.");
          SalesLineLocal.SETRANGE("Location Code", "Location Code");

          //SalesLineLocal.SETFILTER("Qty. to Ship", '>0');

          REPEAT

              TempSalesLine.INIT;

                     QtyPacked := QtyPacked - QtyNeedToPack;
                     TempSalesLine.MODIFY;
                   END;
                END;
              END;
            END;
          UNTIL (SalesLineLocal.NEXT = 0) OR (QtyPacked <= 0);
