    LOCAL PROCEDURE CreateDSCode@1000000002();
    BEGIN
      ShiptoAddress.INIT;
      ShiptoAddress.VALIDATE("Customer No.",Customer."No.");
      ShiptoAddress.VALIDATE(Code,'DS');
      ShiptoAddress."Packing Rule Code" := 'Default';
      ShiptoAddress.Name := '';
      ShiptoAddress.INSERT(TRUE);;
    END;

    LOCAL PROCEDURE CreateShiptoDivision@1000000003();
    BEGIN
      ShiptoAddressDivision.INIT;
      ShiptoAddressDivision.VALIDATE("Customer No.",Customer."No.");
      ShiptoAddressDivision.VALIDATE("Ship-to Code",ShiptoAddress.Code);
      ShiptoAddressDivision.VALIDATE("Shortcut Dimension 5 Code",CustomerDivision."Shortcut Dimension 5 Code");
      ShiptoAddressDivision.VALIDATE("Shipping Advice",CustomerDivision."Shipping Advice");
      ShiptoAddressDivision.VALIDATE("Location Code",CustomerDivision."Location Code");
      IF ShippingAgent.GET(CustomerDivision."Shipping Agent Code") THEN BEGIN
        ShiptoAddressDivision.VALIDATE("Shipping Agent Code",CustomerDivision."Shipping Agent Code");
        ShiptoAddressDivision.VALIDATE("E-Ship Agent Service",CustomerDivision."E-Ship Agent Service");
      END;
      ShiptoAddressDivision.VALIDATE("Shipping Payment Type",CustomerDivision."Shipping Payment Type");
      ShiptoAddressDivision.VALIDATE("Free Freight",CustomerDivision."Free Freight");
      ShiptoAddressDivision.INSERT(TRUE);
    END;

    BEGIN
