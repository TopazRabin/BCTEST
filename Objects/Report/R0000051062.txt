      Location@1000000000 : Record 14;
      WhseCreatePick@1000000002 : Codeunit 5778;
      QtyAvailtoPick@1000000001 : Decimal;
    BEGIN
      // <TPZ1044>
      IF Location.GET(TempSalesLine2."Location Code") AND
         (TempSalesLine2.Type = TempSalesLine2.Type::Item) AND
         (TempSalesLine2.Quantity <> 0)
      THEN BEGIN
        IF NOT Location."Directed Put-away and Pick" THEN
          EXIT(TempSalesLine2.Quantity - TempSalesLine2."Qty. to Ship")
        ELSE
          IF TempSalesLine2."Qty. to Ship" = 0 THEN BEGIN
            QtyAvailtoPick := WhseCreatePick.QtyAvailtoPick(TempSalesLine2."No.",TempSalesLine2."Location Code");
            IF QtyAvailtoPick < (TempSalesLine2.Quantity-TempSalesLine2."Quantity Shipped") THEN //TPZ1749
              EXIT(TempSalesLine2.Quantity-TempSalesLine2."Quantity Shipped" - QtyAvailtoPick); //TPZ1749
          END ELSE
            EXIT(TempSalesLine2.Quantity - TempSalesLine2."Qty. to Ship");
      END;
