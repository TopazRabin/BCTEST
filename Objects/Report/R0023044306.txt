    LOCAL PROCEDURE setupReport@1000000010();
    VAR
      lcuInvMgmt@1000000000 : Codeunit 23044300;
      trecTopLevel@1000000001 : TEMPORARY Record 23044301;
    BEGIN
      // Rolls up totals and does general report setup


      recCountHeader.GET(recInitSheetLine.GETFILTER("Count No."));
      IF (recCountHeader."Inventory As of Date"<>0D) AND
         (recInitSheetLine.GETFILTER("Date Filter")='')
      THEN
        recInitSheetLine.SETRANGE("Date Filter", 0D, recCountHeader."Inventory As of Date");




      IF( ISSERVICETIER ) THEN BEGIN
        bShowWIP := (optShow = optShow::"WIP Difference Only");
      END;


      IF bShowWIP THEN BEGIN
        recInitSheetLine.SETFILTER("Prod. Order No.", '<>%1', '');
        txtReportTitle := tcTitleWIP;
      END ELSE BEGIN
       txtReportTitle := tcTitlePhys;
      END;

      IF recInitSheetLine.GETFILTER("Date Filter")<>'' THEN BEGIN
        dtAsOfDate := recInitSheetLine.GETRANGEMAX("Date Filter");
        trecLineSummary.SETRANGE("Date Filter", 0D, dtAsOfDate);
      END;//if

      lcuInvMgmt.getInventoryDifference(recInitSheetLine, trecLineSummary, bShowWIP,
       FALSE, //no grouping by lot
       FALSE // Group by UOM //<IW author="MH" date="7/22/2015" issue="TFS1895" />
      );

      IF optSorting<>optSorting::"Bin Code" THEN BEGIN
        cuInvCountMgmt.rollUpBinLines(trecLineSummary, trecTopLevel);
        trecLineSummary.DELETEALL;
        trecTopLevel.FINDSET;
        REPEAT
          trecLineSummary := trecTopLevel;
          trecLineSummary."Bin Code" := '';
          trecLineSummary.INSERT;
        UNTIL trecTopLevel.NEXT = 0;
      END;//if sort by bin

      setCountStats(lcuInvMgmt);


      IF NOT (bShowZeroDiff OR bShowOnlySNIssues) THEN
        trecLineSummary.SETFILTER("Difference Qty. (Base)", '<>%1', 0);

      recMainLoop.SETRANGE(Number, 1, trecLineSummary.COUNT);

      setSorting;

      IF NOT trecLineSummary.FIND('-') THEN
        ERROR(tcErrNoDiff);
    END;

    LOCAL PROCEDURE getUOM@1000000000() : Code[10];
    BEGIN
      // Display UOM from bin if UOM is set
      IF trecLineSummary."Bin Code"<>'' THEN BEGIN
        recBinContent.SETRANGE("Location Code", trecLineSummary."Location Code");
        recBinContent.SETRANGE("Bin Code", trecLineSummary."Bin Code");

      EXIT('');
    END;

    PROCEDURE setSorting@1000000009();
    BEGIN
      trecLineSummary.ASCENDING := FALSE;

      CASE optSorting OF
        optSorting::Amount: trecLineSummary.SETCURRENTKEY("Difference Amount (ABS)");
        optSorting::Quantity: trecLineSummary.SETCURRENTKEY("Difference Qty. (ABS) (Base)");

        optSorting::"Item Number":
        BEGIN
          trecLineSummary.ASCENDING := TRUE;
          trecLineSummary.SETCURRENTKEY("Item No.");

      //PP-specific code start
      {
          IF trecLineSummary.FINDSET THEN
            REPEAT
              trecLineSummary."Shelf No." := COPYSTR(trecLineSummary."Item No.",1,19) + '-';
              trecLineSummary.MODIFY;
            UNTIL trecLineSummary.NEXT=0;
          trecLineSummary.SETCURRENTKEY("Shelf No.");
      }
      //PP-specific code end

        END;//if
        optSorting::"Production Order": BEGIN
          trecLineSummary.SETCURRENTKEY("Prod. Order No.");
          trecLineSummary.ASCENDING := TRUE;
        END;//ProdOrder

        optSorting::"Bin Code": BEGIN
          trecLineSummary.SETCURRENTKEY("Bin Code", "Shelf No.");
          trecLineSummary.ASCENDING := TRUE;
        END;//bin

        optSorting::"Item Description": BEGIN
          // this is a bit of a hack to allow sorting by description,
          //   since the count lines don't have a desc field for performance reasons (to reduce memory usage on client)
          trecLineSummary.ASCENDING := TRUE;
          IF trecLineSummary.FINDSET THEN
            REPEAT
              trecLineSummary."Shelf No." := COPYSTR(getDescription,1,40);
              trecLineSummary.MODIFY;
            UNTIL trecLineSummary.NEXT=0;
          trecLineSummary.SETCURRENTKEY("Shelf No.");
        END;//desc

      END;//case
    END;

    LOCAL PROCEDURE setCountStats@1000000005(VAR pcuInvCountMgmt@1000000005 : Codeunit 23044300);
    VAR
      ldLastDiffQty@1000000000 : Decimal;
      ldLastDiffAmt@1000000001 : Decimal;
      ldLastAbsDiffQty@1000000002 : Decimal;
      ldLastAbsDiffAmt@1000000003 : Decimal;
      ldLastInvValue@1000000004 : Decimal;
    BEGIN
      // Set totals

      IF recCountHeader.Active THEN BEGIN
        pcuInvCountMgmt.getLastDifferences(ldLastDiffQty,ldLastDiffAmt,ldLastAbsDiffQty,ldLastAbsDiffAmt,ldLastInvValue);

        recCountHeader."Last Reconcilliation Report" := CURRENTDATETIME;
        recCountHeader."Reported By" := USERID;
        recCountHeader."Last Difference Amt." := ldLastDiffAmt;
        recCountHeader."Last Difference Qty." := ldLastDiffQty;
        recCountHeader."Last Abs. Difference Amt." := ldLastAbsDiffAmt;
        recCountHeader."Last Abs. Difference Qty." := ldLastAbsDiffQty;

        IF ldLastInvValue<>0 THEN
          recCountHeader."Last Difference Pct." := ldLastDiffAmt * 100.0 / ldLastInvValue
        ELSE
          IF ldLastDiffAmt<>0 THEN
            recCountHeader."Last Difference Pct." := 99999
          ELSE
            recCountHeader."Last Difference Pct." := 0;


        recCountHeader.MODIFY;
        COMMIT;
      END;//if
    END;
