    PROCEDURE CalcForecast@1240030001();
    VAR
      ForecastAdjust@1240020001 : Record 14002707;
      ForecastAdjust2@1240020000 : Record 14002707;
      i@1240020002 : Integer;
    BEGIN
      CLEAR(Forecast);
      CLEAR(ForcStartingDateArray);
      CLEAR(ForcEndingDateArray);
      CLEAR(ForcArray);
      CLEAR(ForcAdjArray);
      CLEAR(NewAdjArray);
      Adjust := FALSE;

      Location.GET("Procurement Unit"."Location Code");
      AdvForSetup.GET;

      Forecast.CalcPeriodStartingDates(Location,AdvForSetup."Periods to Forecast");
      Forecast.CalcForecast("Procurement Unit",AdvForSetup."Periods to Forecast",'',FALSE,TRUE);
      Forecast.CalcForecastAdj("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcCustForecast("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcInvPlusExpRec("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcDemand("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcConsumed("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcInventory("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcNetTransfers("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcPurchOrders("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcPriorYear("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.GetComp(AdvForSetup."Periods to Forecast");
      Forecast.CalcProdOrders("Procurement Unit",AdvForSetup."Periods to Forecast");
      Forecast.CalcComp("Procurement Unit",AdvForSetup."Periods to Forecast");

      ForecastAdjust.RESET;
      IF ForecastAdjust.FIND('+') THEN
        EntryNo := ForecastAdjust."Entry No."
      ELSE
        EntryNo := 0;

      ForecastAdjust.SETCURRENTKEY("Item No.","Variant Code","Location Code","Starting Date");
      ForecastAdjust2.SETCURRENTKEY("Item No.","Variant Code","Location Code","Starting Date");
      i := 0;
      REPEAT
        i := i + 1;
        ForcStartingDateArray[i] := Forecast.GetPeriodStartingDate(i);
        ForcEndingDateArray[i] := Forecast.GetPeriodEndingDate(i);
        IF ForcStartingDateArray[i] = StartingDate THEN
          Adjust := TRUE;
        IF Adjust THEN BEGIN
          ForecastAdjust.SETRANGE("Item No." , "Procurement Unit"."Item No.");
          IF "Procurement Unit"."Summarize Variant Detail" THEN
            ForecastAdjust.SETRANGE("Variant Code")
          ELSE
            ForecastAdjust.SETRANGE("Variant Code","Procurement Unit"."Variant Code");
          ForecastAdjust.SETFILTER("Location Code", "Procurement Unit"."Location Code");
          ForecastAdjust.SETRANGE("Starting Date", ForcStartingDateArray[i],ForcEndingDateArray[i]);
          ForecastAdjust.SETRANGE("Frozen Forecast", TRUE);
          IF NOT ForecastAdjust.ISEMPTY THEN BEGIN
            ForecastAdjust.CALCSUMS(ForecastAdjust.Quantity);
            ForcArray[i] := ForecastAdjust.Quantity;
          END ELSE
          ForcArray[i] := Forecast.GetPeriodForecastUsage(i);

          ForecastAdjust2.RESET;
          ForecastAdjust2.SETCURRENTKEY(
            "Item No.","Variant Code","Location Code","Starting Date",
            "Promotion Code");
          ForecastAdjust2.SETRANGE("Item No.","Procurement Unit"."Item No.");
          ForecastAdjust2.SETRANGE("Variant Code","Procurement Unit"."Variant Code");
          ForecastAdjust2.SETRANGE("Location Code","Procurement Unit"."Location Code");
          ForecastAdjust2.SETRANGE("Starting Date",ForcStartingDateArray[i],ForcEndingDateArray[i]);
          ForecastAdjust2.SETRANGE("Frozen Forecast", FALSE);
          ForecastAdjust2.SETFILTER("Forecast Summary Code",'=%1','');
          ForecastAdjust2.SETFILTER("Promotion Code",'=%1','');
          ForecastAdjust2.CALCSUMS(Quantity);
          ForcAdjArray[i] := ForecastAdjust2.Quantity;

          NewAdjArray[i] := ROUND(ForcArray[i] * (PercentAdjust/100),1.0);

          ForecastAdjust.RESET;
          ForecastAdjust.INIT;
          ForecastAdjust."Location Code" := "Procurement Unit"."Location Code";
          ForecastAdjust."Item No." := "Procurement Unit"."Item No.";
          ForecastAdjust."Variant Code" := "Procurement Unit"."Variant Code";
          ForecastAdjust."Starting Date" := ForcStartingDateArray[i];
          ForecastAdjust.Quantity := NewAdjArray[i] - ForcAdjArray[i];
          ForcTotalArray[i] := NewAdjArray[i] + ForcArray[i];
          IF UpdateRecords AND (ForecastAdjust.Quantity <> 0)
          THEN BEGIN
            EntryNo := EntryNo + 1;
            ForecastAdjust."Entry No." := EntryNo;
            ForecastAdjust.INSERT(TRUE);
          END;
        END;
        IF ForcEndingDateArray[i] = EndingDate THEN
          Adjust := FALSE;
      UNTIL i = AdvForSetup."Periods to Calculate";
    END;

    BEGIN
    END.
