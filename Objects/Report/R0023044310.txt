    LOCAL PROCEDURE checkLine@1000000006() bLineValid : Boolean;
    VAR
      lrecGenProdGrp@1000000000 : Record 251;
      lrecInvProdGrp@1000000001 : Record 94;
      lrecItemUOM@1000000002 : Record 5404;
      lrecProdGroup@1000000003 : Record 5723;
      lrecLocation@1000000004 : Record 14;
    BEGIN
      bLineValid := TRUE;

      // check quantity
      IF recJnlLine.Quantity = 0 THEN
        bLineValid := insertError(tcNoQuantity);

      // check on-hand if required
      bLineValid := bLineValid AND checkOnHand;

      // check posting groups
      IF NOT lrecGenProdGrp.GET(recJnlLine."Gen. Prod. Posting Group") THEN
        bLineValid := insertError(tcNoGenProdGrp);

      IF NOT lrecInvProdGrp.GET(recJnlLine."Inventory Posting Group") THEN
        bLineValid := insertError(tcNoInvGrp);

      // Unit of measure
      IF recJnlLine."Unit of Measure Code" = '' THEN
        bLineValid := insertError(tcNoUOM)
      ELSE IF NOT lrecItemUOM.GET(recJnlLine."Item No.", recJnlLine."Unit of Measure Code") THEN
        bLineValid := insertError(tcBadUOM);

      // item tracking lines
      IF recItem."Item Tracking Code" <> '' THEN
        bLineValid := bLineValid AND checkItemTracking;

      // check blocked
      //<TPZ2270>
      //IF recItem.Blocked THEN
        //bLineValid := insertError(tcBlocked);

      IF recItem.Blocked THEN //TPZ2785
        bLineValid := insertError(tcBlocked);
      //</TPZ2270>

      // check item category/product group
      IF bCheckItemCat AND NOT lrecProdGroup.GET(recJnlLine."Item Category Code", recJnlLine."Product Group Code") THEN
        bLineValid := insertError(tcBadCategory);

      // location code
      IF recInvSetup."Location Mandatory" AND (recJnlLine."Location Code" = '') THEN
        bLineValid := insertError(tcNoLocation);

      IF (recJnlLine."Location Code" <> '') AND NOT lrecLocation.GET(recJnlLine."Location Code") THEN
        bLineValid := insertError(tcBadLocation);
    END;

    LOCAL PROCEDURE insertError@1000000007(tcErrorMsg@1000000000 : Text[80]) : Boolean;
    BEGIN
      IF bShowErrMsg THEN BEGIN
        trecCommentLine."Line No." := trecCommentLine."Line No." + 1;
        trecCommentLine.Comment := tcErrorMsg;
        trecCommentLine.INSERT;
      END;//if

      EXIT(FALSE);
    END;

    LOCAL PROCEDURE checkOnHand@1000000000() bLineValid : Boolean;
    VAR
      ldTotalPrevQty@1000000000 : Decimal;
    BEGIN
      bLineValid := TRUE;
      recItem.CALCFIELDS(Inventory);
      ldTotalPrevQty := getTotalJnlQty;  //<IW author="MH" date="12/5/2007" issue="252653" />

      IF (recJnlLine."Entry Type" = recJnlLine."Entry Type"::Consumption) OR
         (recJnlLine."Entry Type" = recJnlLine."Entry Type"::"Negative Adjmt.") OR
         (recJnlLine."Entry Type" = recJnlLine."Entry Type"::Sale)
      THEN BEGIN
        IF ldTotalPrevQty > recItem.Inventory THEN
          bLineValid := insertError(tcNoInventory);
      END ELSE IF (recJnlLine."Entry Type" = recJnlLine."Entry Type"::"Positive Adjmt.") AND (recJnlLine.Quantity<0) THEN BEGIN
        IF ABS(ldTotalPrevQty) > recItem.Inventory THEN
          bLineValid := insertError(tcNoInventory);
      END;//if
    END;

    LOCAL PROCEDURE checkItemTracking@1000000014() bLineValid : Boolean;
    VAR
      lrecResEntry@1000000000 : Record 337;
      ldQty@1000000001 : Decimal;
      lrecItemTracking@1000000005 : Record 6502;
      ldJnlQty@1000000004 : Decimal;
      lbOutboundTrack@1000000003 : Boolean;
      lbInboundTrack@1000000002 : Boolean;
    BEGIN
      // check reservation entries for tracking lines

      // Don't check for tracking unless required
      ldJnlQty := recJnlLine.Quantity * getSign(recJnlLine);
      lrecItemTracking.GET(recItem."Item Tracking Code");
      lbOutboundTrack := (lrecItemTracking."SN Pos. Adjmt. Outb. Tracking" OR
                          lrecItemTracking."SN Neg. Adjmt. Outb. Tracking" OR
                          lrecItemTracking."Lot Pos. Adjmt. Outb. Tracking" OR
                          lrecItemTracking."Lot Neg. Adjmt. Outb. Tracking");

      lbInboundTrack := (lrecItemTracking."SN Pos. Adjmt. Inb. Tracking" OR
                         lrecItemTracking."SN Neg. Adjmt. Inb. Tracking" OR
                         lrecItemTracking."Lot Pos. Adjmt. Inb. Tracking" OR
                         lrecItemTracking."Lot Neg. Adjmt. Inb. Tracking");

      IF (ldJnlQty = 0) OR
         ((ldJnlQty < 0) AND (NOT lbOutboundTrack)) OR
         ((ldJnlQty > 0) AND (NOT lbInboundTrack))
      THEN
        EXIT;

      lrecResEntry.SETRANGE("Item No.", recJnlLine."Item No.");
      lrecResEntry.SETRANGE("Source Type", DATABASE::"Item Journal Line");
      lrecResEntry.SETRANGE("Source ID", recJnlLine."Journal Template Name");
      lrecResEntry.SETRANGE("Source Batch Name", recJnlLine."Journal Batch Name");
      lrecResEntry.SETRANGE("Source Ref. No.", recJnlLine."Line No.");

      IF lrecResEntry.FIND('-') THEN BEGIN
        REPEAT
          IF (lrecResEntry."Serial No." <> '') OR (lrecResEntry."Lot No." <> '') THEN
            ldQty += lrecResEntry.Quantity;
        UNTIL lrecResEntry.NEXT = 0;
      END;//if

      bLineValid := (ABS(ldQty) = ABS(recJnlLine.Quantity));
      IF NOT bLineValid THEN
        insertError(tcMissingSerialNo);
    END;

    LOCAL PROCEDURE getComment@1000000040() txtComment : Text[80];
    BEGIN
      txtComment := trecCommentLine.Comment;
      trecCommentLine.NEXT;
    END;

    LOCAL PROCEDURE getSign@1000000001(VAR lrecCurJnlLine@1000000000 : Record 83) : Integer;
    BEGIN
      CASE lrecCurJnlLine."Entry Type" OF
        lrecCurJnlLine."Entry Type"::"Negative Adjmt.",
        lrecCurJnlLine."Entry Type"::Consumption,
        lrecCurJnlLine."Entry Type"::Sale: EXIT(-1);

        ELSE
          EXIT(1);
      END;
    END;

    LOCAL PROCEDURE getTotalJnlQty@1000000002() dQty : Decimal;
    VAR
      lrecPrevJnlLine@1000000000 : Record 83;
    BEGIN
      // Calculates quantities for current and previous journal lines

      lrecPrevJnlLine.COPYFILTERS(recJnlLine);
      lrecPrevJnlLine.MARKEDONLY := recJnlLine.MARKEDONLY;

      lrecPrevJnlLine.SETRANGE("Line No.", 0, recJnlLine."Line No.");
      lrecPrevJnlLine.SETRANGE("Item No.", recJnlLine."Item No.");
      lrecPrevJnlLine.SETRANGE("Location Code", recJnlLine."Location Code");
      lrecPrevJnlLine.SETRANGE("Variant Code", recJnlLine."Variant Code");

      IF lrecPrevJnlLine.FIND('-') THEN
        REPEAT
         dQty += (-lrecPrevJnlLine.Quantity * getSign(lrecPrevJnlLine));
        UNTIL lrecPrevJnlLine.NEXT = 0;
    END;
