    BEGIN
      liIndex := 1;

      IF( bIncludeDeviceCode ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="config_id" value="%1" /&gt;', diDeviceConfiguration.Code  );
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcDevice);
        liIndex += 1;
      END;

      IF( bIncludeCompanyName ) THEN BEGIN
        lsBarcodeData := 'company='+COMPANYNAME();
        lsCombinedBarcodeData += STRSUBSTNO('&lt;OPTION key="company" value="%1" /&gt;', COMPANYNAME());
        setBarcode(liIndex,'%CONFIG%'+lsBarcodeData,ltcCompany);
        liIndex += 1;
      END;

      IF( bIncludeWebService ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="webservice" value="%1" /&gt;',  txtWebService);
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcWebService);
        liIndex += 1;
      END;

      IF( bIncludeSOAPURL ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="navsoap" value="%1" /&gt;',  txtDynamnicsNavSoapURL);
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcSOAP);
        liIndex += 1;
      END;

      IF( bIncludeLoginMode ) THEN BEGIN
        liTemp := optLoginType;
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="login_mode" value="%1" /&gt;',  FORMAT( liTemp ));
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcLoginMode);
        liIndex += 1;
      END;

      IF( bIncludeDomain ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="default_domain" value="%1" /&gt;',  txtDefaultDomain);
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcDomain);
        liIndex += 1;
      END;

      IF( bIncludeUser ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="default_user" value="%1" /&gt;',  txtDefaultUser);
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcUser);
        liIndex += 1;
      END;

      IF( bIncludePassword ) THEN BEGIN
        lsBarcodeData := STRSUBSTNO('&lt;OPTION key="default_password" value="%1" /&gt;', txtDefaultPassword);
        lsCombinedBarcodeData += lsBarcodeData;
        setBarcode(liIndex,lsBarcodeData,ltcPassword);
        liIndex += 1;
      END;

      IF( bIncludeReload ) THEN BEGIN
        lsBarcodeData := '&lt;OPTION key="reload" value="1" /&gt;';
        lsCombinedBarcodeData += lsBarcodeData;
      //  lsBarcodeData := 'RELOAD';
        // note: reload is built into the combined automatically
        setBarcode(liIndex,lsBarcodeData,ltcReload);
        liIndex += 1;
      END;

      IF( bIncludeCombined ) THEN BEGIN
        setBarcode(liIndex,lsCombinedBarcodeData,ltcCombined);
        liIndex += 1;
      END;
    END;
