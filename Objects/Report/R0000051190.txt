    LOCAL PROCEDURE InsertTempLP@1000000016();
    VAR
      LPHdr@1000000000 : Record 23044505;
      PalletEntry@1000000001 : Record 50068;
    BEGIN
      LPHdr.RESET;
      LPHdr.SETRANGE("Source No.",trecParam."Document No.");
      IF LPHdr.FINDFIRST THEN REPEAT
        LPHdr.CALCFIELDS("Source No.");
        BOLNo := LPHdr."BOL No.";
        PalletEntry.RESET;
        PalletEntry.SETRANGE("License Plate No.",LPHdr."No.");
        PalletEntry.SETRANGE(Status,PalletEntry.Status::Released);
        IF PalletEntry.FINDFIRST THEN
          BEGIN
            REPEAT
              EntryNo := INCSTR(EntryNo);
              TempLPHdr.RESET;
              TempLPHdr."No." := EntryNo;
              TempLPHdr.INSERT;
              TempLPHdr."Source Document No." := LPHdr."Source No.";
              TempLPHdr."Template Code" := FORMAT(PalletEntry."Entry Type");
              TempLPHdr."Shipped Item Count" := PalletEntry."No of Pallet/Box";
              TempLPHdr."Shipment Gross Weight" := PalletEntry.Weight;
              TempLPHdr.MODIFY;
            UNTIL PalletEntry.NEXT = 0;
          END ELSE
          BEGIN
            EntryNo := INCSTR(EntryNo);
            TempLPHdr.RESET;
            TempLPHdr."No." := EntryNo;
            TempLPHdr.INSERT;
            TempLPHdr."Source Document No." := LPHdr."Source No.";
            TempLPHdr."Template Code" := 'Pallet';
            TempLPHdr."Shipped Item Count" := 1;
            TempLPHdr."Shipment Gross Weight" := LPHdr."Shipment Gross Weight";
            TempLPHdr."Shipment Length" := LPHdr."Shipment Length";
            TempLPHdr."Shipment Height" := LPHdr."Shipment Height";
            TempLPHdr."Shipment Width" := LPHdr."Shipment Width";
            TempLPHdr.MODIFY;
          END;
      UNTIL LPHdr.NEXT = 0;
    END;
