    LOCAL PROCEDURE InsertCommand@1(Barcode@1000000000 : Code[20];Description@1000000001 : Text[250];FieldCaption@1240030002 : Text[250]);
    VAR
      CommandTranslation@1240030000 : Record 14000727;
      OriginalBarcode@1240030001 : Code[20];
    BEGIN
      OriginalBarcode := '';
      IF NOT NoCommandTranslation THEN
        IF STRLEN(Barcode) > 2 THEN
          IF COPYSTR(Barcode,1,2) = '//' THEN BEGIN
            CommandTranslation.RESET;
            CommandTranslation.SETCURRENTKEY("Language ID","Replace Command Code");
            CommandTranslation.SETRANGE("Language ID",GLOBALLANGUAGE);
            CommandTranslation.SETRANGE("Replace Command Code",Barcode);
            IF CommandTranslation.FIND('-') THEN BEGIN
              OriginalBarcode := Barcode;

              Barcode := CommandTranslation."Command Code";
              Description := CommandTranslation.Description;
            END ELSE
              IF OnlyWithCommandTranslation THEN
                EXIT;
          END;

      CustomerTmp."No." := Barcode;
      CustomerTmp.Name := COPYSTR(Description,1,MAXSTRLEN(CustomerTmp.Name));
      CustomerTmp."Name 2" := COPYSTR(FieldCaption,1,MAXSTRLEN(CustomerTmp."Name 2"));
      IF NOT CustomerTmp.INSERT THEN BEGIN
        CustomerTmp.GET(CustomerTmp."No.");
        IF OriginalBarcode <> '' THEN
          MESSAGE(Text084,Barcode,Description,CustomerTmp."No.",CustomerTmp.Name,OriginalBarcode)
        ELSE
          MESSAGE(Text085,Barcode,Description,CustomerTmp."No.",CustomerTmp.Name);
      END;
    END;

      BarcodeFieldCaption[Number] := CustomerTmp."Name 2";
    END;
