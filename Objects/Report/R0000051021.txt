               OnPreDataItem=BEGIN
                               SETRANGE("Shortcut Dimension 5 Code",ShortcutDim5Code);
                             END;

               OnAfterGetRecord=BEGIN
                                  FreightAmount := 0;
                                  RebateAmount := 0;
                                  MiscAmount := 0;

                                  ActualCost := -CostCalcMgt.CalcCustLedgActualCostLCY(CustLedgerEntry);
                                  //AdjustedCost := -CostCalcMgt.CalcCustLedgAdjmtCostLCY(CustLedgerEntry);

                                  CASE "Document Type" OF
                                    "Document Type"::Invoice:
                                      BEGIN
                                        SalesInvLine.RESET;
                                        SalesInvLine.SETRANGE("Document No.","Document No.");
                                        SalesInvLine.SETFILTER(Type,'%1|%2',SalesInvLine.Type::"G/L Account",SalesInvLine.Type::Resource);
                                        IF SalesInvLine.FINDSET THEN
                                          REPEAT
                                            IF SalesInvLine.Type = SalesInvLine.Type::"G/L Account" THEN
                                              CASE SalesInvLine."No." OF
                                                SalesSetup."Freight G/L Acc. No.":
                                                  FreightAmount := FreightAmount + SalesInvLine.Amount;
                                                SalesSetup."G/L Rebate Account No.":
                                                  RebateAmount := RebateAmount + SalesInvLine.Amount;
                                                ELSE
                                                  MiscAmount := MiscAmount + SalesInvLine.Amount;
                                              END;

                                            IF SalesInvLine.Type = SalesInvLine.Type::Resource THEN BEGIN
                                              IF Resource.GET(SalesInvLine."No.") AND (Resource."Charge Type" = Resource."Charge Type"::Freight) THEN
                                                FreightAmount := FreightAmount + SalesInvLine.Amount
                                              ELSE
                                                MiscAmount := MiscAmount + SalesInvLine.Amount;
                                            END;
                                          UNTIL SalesInvLine.NEXT = 0;
                                       END;
                                    "Document Type"::"Credit Memo":
                                      BEGIN
                                        SalesCrMemoLine.RESET;
                                        SalesCrMemoLine.SETRANGE("Document No.","Document No.");
                                        SalesCrMemoLine.SETFILTER(Type,'%1|%2',SalesCrMemoLine.Type::"G/L Account",SalesCrMemoLine.Type::Resource);
                                        IF SalesCrMemoLine.FINDSET THEN
                                          REPEAT
                                            IF SalesCrMemoLine.Type = SalesCrMemoLine.Type::"G/L Account" THEN
                                              CASE SalesInvLine."No." OF
                                                SalesSetup."Freight G/L Acc. No.":
                                                  FreightAmount := FreightAmount + SalesCrMemoLine.Amount;
                                                SalesSetup."G/L Rebate Account No.":
                                                  RebateAmount := RebateAmount + SalesCrMemoLine.Amount;
                                                ELSE
                                                  MiscAmount := MiscAmount + SalesCrMemoLine.Amount;
                                              END;

                                            IF SalesCrMemoLine.Type = SalesCrMemoLine.Type::Resource THEN BEGIN
                                              IF Resource.GET(SalesCrMemoLine."No.") AND (Resource."Charge Type" = Resource."Charge Type"::Freight) THEN
                                                FreightAmount := FreightAmount + SalesCrMemoLine.Amount
                                              ELSE
                                                MiscAmount := MiscAmount + SalesCrMemoLine.Amount;
                                            END;

                                          UNTIL SalesCrMemoLine.NEXT = 0;
                                        FreightAmount := -FreightAmount;
                                        RebateAmount := -RebateAmount;
                                        MiscAmount := -MiscAmount;
                                      END;
                                  END;

                                  SalesAmountExclGL := "Sales (LCY)" - (FreightAmount + RebateAmount + MiscAmount);
                                  //ProfitAmount :=  SalesAmountExclGL - (ActualCost + AdjustedCost);
                                  ProfitAmount :=  SalesAmountExclGL - ActualCost;
                                  IF SalesAmountExclGL <> 0 THEN
                                    ProfitPct := ROUND(100 * ProfitAmount / SalesAmountExclGL,0.1)
                                  ELSE
                                    ProfitPct := 0;
                                END;

               ReqFilterFields=Posting Date;
               DataItemLink=Sell-to Customer No.=FIELD(No.) }
