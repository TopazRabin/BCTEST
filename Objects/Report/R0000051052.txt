    PROCEDURE CreateTempPackageLineDetail@1000000001(VAR CurrPackageLine@1000000000 : Record 14000702);
    VAR
      QtyPacked@1000000003 : Decimal;
      QtyNeedToPack@1000000002 : Decimal;
      PackageLine3@1000000001 : Record 14000702;
      TransferLineLocal@1000000004 : Record 5741;
    BEGIN
      TempPackageLineDetail.RESET;
      TempPackageLineDetail.DELETEALL;

      IF CurrPackageLine."Source Type" = DATABASE::"Sales Header" THEN BEGIN
        TempPackageLineDetail.INIT;
        TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
        TempPackageLineDetail.INSERT;
        EXIT;
      END;

      IF CurrPackageLine."Source Type" = DATABASE::"Purchase Header" THEN BEGIN
        TempPackageLineDetail.INIT;
        TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
        TempPackageLineDetail.INSERT;
        EXIT;
      END;

      WITH CurrPackageLine DO BEGIN
        IF ("Source Type" = DATABASE::"Transfer Header") AND (Type = Type::Item)THEN BEGIN
          PackageLine3.RESET;
          PackageLine3.SETFILTER("Package No.", '<%1', "Package No.");
          PackageLine3.SETRANGE("Source Type", "Source Type");
          PackageLine3.SETRANGE("Source Subtype", "Source Subtype");
          PackageLine3.SETRANGE("Source ID", "Source ID");
          PackageLine3.SETRANGE("No.", "No.");

          PackageLine3.SETRANGE("Location Code", "Location Code");

          CreateTempTransferLine(CurrPackageLine, PackageLine3);

          PackageLine3.RESET;
          PackageLine3.SETRANGE("Package No.", "Package No.");
          PackageLine3.SETRANGE("Source Type", "Source Type");
          PackageLine3.SETRANGE("Source Subtype", "Source Subtype");
          PackageLine3.SETRANGE("Source ID", "Source ID");
          PackageLine3.SETRANGE("No.", "No.");

          PackageLine3.SETRANGE("Location Code", "Location Code");

          PackageLine3.SETFILTER("Line No.", '<%1', "Line No.");
          CreateTempTransferLine(CurrPackageLine, PackageLine3);

          //////////////////////////////////////////////////////////////////////
          QtyPacked := Quantity;
          TransferLineLocal.RESET;
          TransferLineLocal.SETRANGE("Document No.", "Source ID");
          TransferLineLocal.SETRANGE("Item No.", "No.");
          //SalesLineLocal.SETFILTER("Qty. to Ship", '>0');
          TransferLineLocal.SETRANGE("Transfer-from Code", "Location Code");

          IF TransferLineLocal.FIND('-') THEN
          REPEAT
            IF TempTransferLine.GET(TransferLineLocal."Document No.",
                                    TransferLineLocal."Line No.") = FALSE THEN BEGIN
               TempTransferLine.INIT;
               TempTransferLine.TRANSFERFIELDS(TransferLineLocal);
               TempTransferLine."Qty. to Ship" := 0;
               TempTransferLine."Qty. to Ship (Base)" := 0;
               TempTransferLine.INSERT;
            END;
            QtyNeedToPack := TransferLineLocal."Outstanding Quantity" - TempTransferLine."Qty. to Ship";

            IF QtyNeedToPack > 0 THEN BEGIN
              TempPackageLineNo := TempPackageLineNo + 10;
              TempPackageLineDetail.INIT;
              TempPackageLineDetail.TRANSFERFIELDS(CurrPackageLine);
              TempPackageLineDetail."Line No." := TempPackageLineNo;

              IF QtyNeedToPack >= QtyPacked THEN BEGIN
                TempPackageLineDetail.Quantity := QtyPacked;
                QtyPacked := 0;
              END ELSE BEGIN
                TempPackageLineDetail.Quantity := QtyNeedToPack;
                QtyPacked := QtyPacked - QtyNeedToPack;
              END;

              TempPackageLineDetail."Quantity (Base)" := TempPackageLineDetail.Quantity *
                                                            TempPackageLineDetail."Qty. per Unit of Measure";
              TempPackageLineDetail."Source Line No." := TransferLineLocal."Line No.";
              TempPackageLineDetail.INSERT;

              TempTransferLine."Qty. to Ship" := TempTransferLine."Qty. to Ship" +
                                                        TempPackageLineDetail.Quantity;
              TempTransferLine."Qty. to Ship (Base)" := TempTransferLine."Qty. to Ship (Base)" +
                                                        TempPackageLineDetail."Quantity (Base)";
              TempTransferLine.MODIFY;

            END;
          UNTIL (TransferLineLocal.NEXT = 0) OR (QtyPacked <= 0);

          IF QtyPacked > 0 THEN
            MESSAGE(Text00001, "No.");

        END;
      END;
    END;

    PROCEDURE CreateTempTransferLine@1000000000(VAR CurrPackageLine@1000000003 : Record 14000702;VAR PackageLine3@1000000000 : Record 14000702);
    VAR
      QtyPacked@1000000002 : Decimal;
      QtyNeedToPack@1000000001 : Decimal;
      TransferLineLocal@1000000004 : Record 5741;
    BEGIN
      WITH CurrPackageLine DO
      BEGIN
        IF PackageLine3.FIND('-') THEN
        REPEAT
          QtyPacked := PackageLine3.Quantity;
          TransferLineLocal.RESET;
          TransferLineLocal.SETRANGE("Document No.", "Source ID");
          TransferLineLocal.SETRANGE("Item No.", "No.");
          TransferLineLocal.SETRANGE("Transfer-from Code", "Location Code");

          //SalesLineLocal.SETFILTER("Qty. to Ship", '>0');
          IF TransferLineLocal.FIND('-') THEN
          REPEAT
            IF TempTransferLine.GET(TransferLineLocal."Document No.", TransferLineLocal."Line No.") = FALSE THEN BEGIN
              TempTransferLine.INIT;
              TempTransferLine.TRANSFERFIELDS(TransferLineLocal);

              IF TempTransferLine."Qty. to Ship" > QtyPacked THEN BEGIN
                TempTransferLine."Qty. to Ship" := QtyPacked;
                TempTransferLine."Qty. to Ship (Base)" := QtyPacked * PackageLine3."Qty. per Unit of Measure";
                QtyPacked := 0;
              END ELSE BEGIN
                QtyPacked := QtyPacked - TempTransferLine."Qty. to Ship";
              END;
              TempTransferLine.INSERT;
            END ELSE BEGIN
              IF TempTransferLine."Qty. to Ship" < TransferLineLocal."Qty. to Ship" THEN BEGIN
                QtyNeedToPack := TransferLineLocal."Outstanding Quantity" - TempTransferLine."Qty. to Ship";
                IF QtyNeedToPack > 0 THEN BEGIN
                   IF QtyNeedToPack >= QtyPacked THEN BEGIN
                     TempTransferLine."Qty. to Ship" := TempTransferLine."Qty. to Ship" + QtyPacked;
                     QtyPacked := 0;
                     TempTransferLine.MODIFY;
                   END ELSE BEGIN
                     TempTransferLine."Qty. to Ship" := TempTransferLine."Qty. to Ship" + QtyNeedToPack;
                     QtyPacked := QtyPacked - QtyNeedToPack;
                     TempTransferLine.MODIFY;
                   END;
                END;
              END;
            END;
          UNTIL (TransferLineLocal.NEXT = 0) OR (QtyPacked <= 0);
        UNTIL PackageLine3.NEXT = 0;
      END;
    END;
