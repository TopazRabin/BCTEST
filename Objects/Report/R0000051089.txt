                                  BaseUPC := '';

                                  InnerUPCFound := FALSE;
                                  MasterUPCFound:= FALSE;
                                  ItemCrossReference.RESET;

                                  ItemCrossReference.SETFILTER("Cross-Reference No.",'<>%1','');
                                  IF ItemCrossReference.FINDSET THEN BEGIN
                                    REPEAT
                                      IF ItemCrossReference."Unit of Measure" = Item."Base Unit of Measure" THEN BEGIN
                                        IF BaseUPCFound = FALSE THEN
                                          BaseUPC := ItemCrossReference."Cross-Reference No.";
                                         BaseUPCFound := TRUE;
                                      END ELSE IF ItemCrossReference."Unit of Measure" = 'I' THEN BEGIN
                                        IF InnerUPCFound = FALSE THEN
                                          InnerUPC := ItemCrossReference."Cross-Reference No.";
                                         InnerUPCFound := TRUE;
                                      END ELSE IF ItemCrossReference."Unit of Measure" = 'M' THEN BEGIN
                                        IF MasterUPCFound = FALSE THEN
                                          MasterUPC := ItemCrossReference."Cross-Reference No.";
                                         MasterUPCFound := TRUE;
                                      END;
                                    UNTIL ItemCrossReference.NEXT=0;
                                  END;
                                END;
                                 }
