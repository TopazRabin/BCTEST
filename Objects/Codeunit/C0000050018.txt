    PROCEDURE UpdateOrderMultiple@1000000000();
    VAR
      Item@1000000000 : Record 27;
      Window@1000000001 : Dialog;
      Counter@1000000002 : Integer;
      ItemUOM@1000000003 : Record 5404;
    BEGIN
      Window.OPEN('#1###### OF #2######');
      Window.UPDATE(2, Item.COUNT);
      Item.FINDSET;
      REPEAT
        Counter += 1;
        Window.UPDATE(1, Counter);
        ItemUOM.SETRANGE("Item No.", Item."No.");
        ItemUOM.SETRANGE(Code, 'M');
        IF ItemUOM.FINDFIRST THEN BEGIN
          Item."Order Multiple" := ItemUOM."Qty. per Unit of Measure";
          Item.MODIFY;
        END;
      UNTIL Item.NEXT = 0
    END;

    BEGIN
    END.
