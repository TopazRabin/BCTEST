    [EventSubscriber(Codeunit,51949,CU448_CalcRunTimeForRecurringJob)]
    PROCEDURE CU448_CalcRunTimeForRecurringJob@1000000006(VAR JobQueueEntry@1000000000 : Record 472;VAR NewRunDateTime@1000000001 : DateTime);
    BEGIN
      // <TPZ70>
      IF JobQueueEntry."Run Monthly" THEN BEGIN
        NewRunDateTime :=
          CREATEDATETIME(CALCDATE('<1M>',TODAY),JobQueueEntry."Starting Time");

      END;
      IF JobQueueEntry."Run Biweekly" THEN BEGIN
        NewRunDateTime :=
          CREATEDATETIME(CALCDATE('<2W>',TODAY),JobQueueEntry."Starting Time");

      END;
    END;

    BEGIN
    END.
  }
