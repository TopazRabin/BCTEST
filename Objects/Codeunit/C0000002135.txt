OBJECT Codeunit 2135 O365 Setup Email
{
  OBJECT-PROPERTIES
  {
    Date=11/22/17;
    Time=12:00:00 PM;
    Version List=NAVW111.00;
  }
  PROPERTIES
  {
    OnRun=BEGIN
            SetupEmail(FALSE);
          END;

  }
  CODE
  {
    VAR
      MailNotConfiguredErr@1000 : TextConst 'ENU=An email account must be configured to send emails.;ESM=Se debe configurar una cuenta de correo electr¢nico para enviar correos electr¢nicos.;FRC=Un compte de messagerie doit ˆtre configur‚ pour envoyer des courriers ‚lectroniques.;ENC=An email account must be configured to send emails.';
      ClientTypeManagement@1001 : Codeunit 4;

    [External]
    PROCEDURE SetupEmail@15(ForceSetup@1000 : Boolean);
    BEGIN
      IF ForceSetup OR (NOT SMTPEmailIsSetUp) THEN BEGIN
        IF ClientTypeManagement.GetCurrentClientType = CLIENTTYPE::Tablet THEN BEGIN
          IF PAGE.RUNMODAL(PAGE::"BC O365 Email Setup Wizard") <> ACTION::LookupOK THEN
            ERROR('');
        END ELSE
          PAGE.RUNMODAL(PAGE::"O365 Email Setup Wizard");
        IF NOT SMTPEmailIsSetUp THEN
          ERROR(MailNotConfiguredErr);
      END;
    END;

    PROCEDURE SMTPEmailIsSetUp@1() : Boolean;
    VAR
      SMTPMailSetup@1001 : Record 409;
      MailManagement@1002 : Codeunit 9520;
    BEGIN
      IF NOT MailManagement.IsSMTPEnabled THEN
        EXIT(FALSE);

      IF SMTPMailSetup.FIND THEN
        EXIT((SMTPMailSetup."User ID" <> '') AND (SMTPMailSetup.GetPassword <> ''));
    END;

    BEGIN
    END.
  }
}
