OBJECT Codeunit 51400 Before Cutoff
{
  OBJECT-PROPERTIES
  {
    Date=07/01/20;
    Time=12:00:00 AM;
    Modified=Yes;
    Version List=TOP190,TPZHOTFIX;
  }
  PROPERTIES
  {
    OnRun=BEGIN
            // Deactivate change log

            ChangeLogSetup."Change Log Activated" := FALSE;

            // Turn on direct posting for bank account G/L accounts

            GLAccount.GET('101110');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('101120');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('101130');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            // Turn on direct posting for AR G/L accounts

            GLAccount.GET('111000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('112000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('113000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            // Turn on direct posting for A/P G/L accounts

            GLAccount.GET('211000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('212000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('213000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            // Turn on direct posting for inventory G/L accounts

            GLAccount.GET('121000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('122000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            GLAccount.GET('123000');
            GLAccount."Direct Posting" := TRUE;
            GLAccount.MODIFY;

            // Change Inventory Adjmt. Account in General Posting Setup for inventory import

            GeneralPostingSetup.RESET;
            GeneralPostingSetup.SETRANGE("Gen. Prod. Posting Group",'E');
            GeneralPostingSetup.MODIFYALL("Inventory Adjmt. Account",'122000');

            GeneralPostingSetup.RESET;
            GeneralPostingSetup.SETRANGE("Gen. Prod. Posting Group",'I');
            GeneralPostingSetup.MODIFYALL("Inventory Adjmt. Account",'123000');

            GeneralPostingSetup.RESET;
            GeneralPostingSetup.SETRANGE("Gen. Prod. Posting Group",'L');
            GeneralPostingSetup.MODIFYALL("Inventory Adjmt. Account",'121000');

            // Turn off External Document No. Mandatory

            SalesSetup.GET;
            SalesSetup."Ext. Doc. No. Mandatory" := FALSE;
            SalesSetup.MODIFY;

            PurchSetup.GET;
            PurchSetup."Ext. Doc. No. Mandatory" := FALSE;
            PurchSetup.MODIFY;

            // Turn off Validate Cust. Mfr. Rep. Exc.

            SalesSetup.GET;
            SalesSetup."Validate Cust. Mfr. Rep. Exc." := FALSE;
            SalesSetup.MODIFY;

            // Turn off Bin Capacity Check on Locations 1, 25, 30

            Location.RESET;
            Location.SETFILTER(Code,'1|25|30');
            Location.MODIFYALL("Bin Capacity Policy",Location."Bin Capacity Policy"::"Never Check Capacity");

            // Turn on Manual Nos. for Sales Quote, Sales Order, and Sales Return Order No. Series

            NoSeries.RESET;
            NoSeries.SETFILTER(Code,'S-QUO|S-RETORD|P-ORD');
            NoSeries.MODIFYALL("Manual Nos.",TRUE);

            MESSAGE('DONE');
          END;

  }
  CODE
  {
    VAR
      ChangeLogSetup@1000000001 : Record 402;
      GLAccount@1000000000 : Record 15;
      GeneralPostingSetup@1000000004 : Record 252;
      SalesSetup@1000000002 : Record 311;
      PurchSetup@1000000003 : Record 312;
      Location@1000000005 : Record 14;
      NoSeries@1000000006 : Record 308;

    BEGIN
    END.
  }
}
