OBJECT Codeunit 50054 Item Validation Process
{
  OBJECT-PROPERTIES
  {
    Date=07/01/20;
    Time=12:00:00 AM;
    Modified=Yes;
    Version List=TPZ000.00.00;
  }
  PROPERTIES
  {
    OnRun=BEGIN
            {
            IF JobReportsEmail.GET(51125) THEN
                REPORT.SAVEASEXCEL(51125,JobReportsEmail."File Path"+'ItemValidations.xlsx');
              MESSAGE('Done');
            }

            IF JobReportsEmail.GET(51131) THEN BEGIN
              ItemValidations.ApplyFilters(TRUE,FALSE,TRUE);
              ItemValidations.SAVEASEXCEL(JobReportsEmail."File Path"+'Un posted journal with blocked items.xlsx');
              CLEAR(ItemValidations);
              ItemValidations.ApplyFilters(FALSE,TRUE,TRUE);
              ItemValidations.SAVEASEXCEL(JobReportsEmail."File Path"+'Open SOs with blocked items.xlsx');
              CLEAR(ItemValidations);
              ItemValidations.ApplyFilters(TRUE,TRUE,FALSE);
              ItemValidations.SAVEASEXCEL(JobReportsEmail."File Path"+'Open POs and transfers with blocked items.xlsx');
            END;
          END;

  }
  CODE
  {
    VAR
      JobReportsEmail@1000000000 : Record 50015;
      ItemValidations@1000000001 : Report 51131;

    BEGIN
    END.
  }
}
