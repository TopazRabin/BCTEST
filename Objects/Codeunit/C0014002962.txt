{
  OBJECT-PROPERTIES
  {
    Date=10/27/10;
    Time=12:00:00 PM;
    Version List=RF1.20.31;
  }
  PROPERTIES
  {
    TableNo=5766;
    OnRun=BEGIN
            WITH Rec DO BEGIN
              WarehouseActivityLine.RESET;
              WarehouseActivityLine.SETRANGE("Activity Type",Type);
              WarehouseActivityLine.SETRANGE("No.","No.");
              WarehouseActivityLine.SETFILTER("Breakbulk No.",'<>0');
              IF NOT WarehouseActivityLine.FIND('-') THEN
                EXIT;

              WarehouseActivityLine.SETRANGE("Breakbulk No.");
              WarehouseActivityLine.FIND('-');
              REPEAT
                IF WarehouseActivityLine."Breakbulk No." <> 0 THEN BEGIN
                  IF WarehouseActivityLine."Qty. to Handle" <> WarehouseActivityLine."Qty. Outstanding"
                  THEN BEGIN
                    WarehouseActivityLine.VALIDATE(
                      "Qty. to Handle",WarehouseActivityLine."Qty. Outstanding");
                    WarehouseActivityLine.MODIFY(TRUE);
                  END;
                END ELSE
                  IF WarehouseActivityLine."Qty. to Handle" <> 0 THEN BEGIN
                    WarehouseActivityLine.VALIDATE("Qty. to Handle",0);
                    WarehouseActivityLine.MODIFY(TRUE);
                  END;
              UNTIL WarehouseActivityLine.NEXT = 0;

              WarehouseActivityLine.FIND('-');

              WhseActivityRegister.RUN(WarehouseActivityLine);
            END;
          END;

  }
  CODE
  {

      WhseActivityRegister@1240030002 : Codeunit 7307;

    BEGIN
    END.
  }
